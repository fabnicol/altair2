179 bool MatriculeInput::checkInput(FLineEdit* l)
180 {
181  bool res = true;
182 
183  QString s = l->text();
184  QList<QString> L;
185 
186  for (auto &&S : s.split(';', QString::SkipEmptyParts))
187  {
188  QList<QString> sublist = S.split('-', QString::SkipEmptyParts);
189 
190  if (sublist.size() != 3)
191  {
192  QMessageBox::warning(nullptr, "Format de la requête " + S,
193  "Utiliser le format suivant : matricule-mois-année.<br>"
194  "Un intervalle peut être utilisé pour le mois et l'année.<br>"
195  "Exemple :<br>\t 1012B-1...7-2011...2013", QMessageBox::Ok);
196  S.clear();
197  }
198 
199  int cut = 0, cut2 = 0;
200  const QString s1 = sublist.at(1);
201  bool range = false;
202 
203  if (s1.contains("..."))
204  {
205  cut = s1.split("...", QString::SkipEmptyParts).at(0).toInt();
206  cut2 = s1.split("...", QString::SkipEmptyParts).at(1).toInt();
207  range = true;
208  }
209 
210  if (! s1.contains(QRegularExpression("[0-9]+(?:\\.\\.\\.[0-9]+|)"))
211  || (range && (cut > 12 || cut < 1) && (cut2 > 12 || cut2 < 1)) || (! range && (s1.toInt() > 12 || s1.toInt() < 1)))
212  {
213  QMessageBox::warning(nullptr, "Format de la requête " + S, "Le mois doit être compris entre 1 et 12.", QMessageBox::Ok);
214  S.clear();
215  }
216 
217  if (! sublist.at(2).contains(QRegularExpression("[0-9]{4}(?:\\.\\.\\.[0-9]{4}|)")) || (! range && sublist.at(2).toInt() < 2007))
218  {
219  QMessageBox::warning(nullptr, "Format de la requête " + S, "L'année doit être numérique, à compter de 2008.", QMessageBox::Ok);
220  S.clear();
221  }
222 
223  if (! S.isEmpty())
224  L << S;
225  else
226  res = false;
227  }
228 
229  l->setText(L.join(";"));
230  return res;
231 }
