227 bool MatriculeInput::checkInput (FLineEdit* l)
228 {
229  bool res = true;
230 
231  QString s = l->text();
232  QList<QString> L;
233 
234  for (auto &&S : s.split (';', QString::SkipEmptyParts))
235  {
236  QList<QString> sublist = S.split ('-', QString::SkipEmptyParts);
237 
238  if (sublist.size() != 3)
239  {
240  QMessageBox::warning (nullptr, "Format de la requête " + S,
241  "Utiliser le format suivant : matricule-mois-année.<br>"
242  "Un intervalle peut être utilisé pour le mois et l'année.<br>"
243  "Exemple :<br>\t 1012B-1...7-2011...2013", QMessageBox::Ok);
244  }
245 
246 
247  if (sublist.at(0).contains(QRegularExpression ("[;:]")))
248  QMessageBox::warning (nullptr, "Format de la requête " + S,
249  "Les matricules doivent être séparés par une virgule.",
250  QMessageBox::Ok);
251 
252  int cut = 0, cut2 = 0, cut3 = 0, cut4 = 0;
253  const QString s1 = sublist.at (1);
254  const QString s2 = sublist.at (2);
255  bool range_month = false;
256  bool range_year = false;
257 
258  if (s1.contains ("..."))
259  {
260  auto l1 = s1.split ("...", QString::SkipEmptyParts);
261  cut = l1.at (0).toInt();
262  cut2 = l1.at (1).toInt();
263  range_month = true;
264  }
265 
266  if (s2.contains ("..."))
267  {
268  auto l2 = s2.split ("...", QString::SkipEmptyParts);
269  cut3 = l2.at (0).toInt();
270  cut4 = l2.at (1).toInt();
271  range_year = true;
272  }
273 
274  if ((range_month && (cut > 12 || cut < 1 || cut2 > 12 || cut2 < 1)) || (! range_month && (s1.toInt() > 12 || s1.toInt() < 1)))
275  {
276  QMessageBox::warning (nullptr, "Format de la requête " + S, "Le mois doit être compris entre 1 et 12.", QMessageBox::Ok);
277  }
278 
279  if ((range_year && ((cut3 % 2000) < 8 || (cut4 % 2000) < 8)) || (! range_year && (s2.toInt() % 2000) < 8))
280  {
281  QMessageBox::warning (nullptr, "Format de la requête " + S,
282  "L'année doit être numérique, à compter de 2008.",
283  QMessageBox::Ok);
284  }
285 
286  if (! S.isEmpty())
287  L << S;
288  else
289  res = false;
290  }
291 
292  l->setText (L.join (";"));
293  return res;
294 }
