124 /* Note :
125  * Windows instantiation request non-recursive abstract class pointer lists, ie,
126  * a list of pointers to FAbstractWidget cannot be a member of FAbstractWidget
127  * This compiles but leads to severe runtime crashes (Qt5.0.2 + mingw-g++4.7 + windows XP or 7)
128  * Both compiles and runs OK under Linux however (Qt5.0.2 + g++4.7 or 4.8 + Ubuntu 13.04).
129  * The following one-member abstract structure works out this intriguing issue that remains poorly understood */

211  static void initH(const QString &hashKey, const QString& value)
212  {
213  if (! Hash::wrapper.isEmpty()
214  && Hash::wrapper.value(hashKey, nullptr) != nullptr)
215  delete(Hash::wrapper[hashKey]);
216 
217  *(Hash::wrapper[hashKey] = new FStringList) = value;

224  static void initH(const QString &hashKey)
225  {
226  // Détruire la valeur allouée de Hash::wrapper[hashKey] si elle existe
227  // et si elle est allouée et réallouer une nouvelle à blanc
228 
229  if (! Hash::wrapper.isEmpty()
230  && Hash::wrapper.value(hashKey, nullptr) != nullptr)
231  delete(Hash::wrapper[hashKey]);
232 
233  Hash::wrapper[hashKey] = new FStringList;
234  }
