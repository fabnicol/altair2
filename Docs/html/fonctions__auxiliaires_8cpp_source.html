212 errorLine_t afficher_environnement_xhl (const info_t& info, const xmlNodePtr cur)
213 {
214 
215  long lineN = 0;
216  LOCK_GUARD
217  cerr << WARNING_HTML_TAG "Fichier analysé " << info.threads->argv[info.fichier_courant] << ENDL;
218 
219  if (cur)
220  {
221  lineN = (long) xmlGetLineNo (cur);
222 
223  if (lineN == 65535 && verbeux
224  && info.ligne_debut.size() > info.NCumAgentXml
225  && info.ligne_fin.size() > info.NCumAgentXml)
226  {
227  cerr << WARNING_HTML_TAG "Entre les lignes "
228  << info.ligne_debut.at (info.NCumAgentXml)[0] + 1
229  << " et "
230  << info.ligne_fin.at (info.NCumAgentXml)[0] << ENDL;
231  }
232  else
233  {
234  if (lineN == -1)
235  {
236  cerr << WARNING_HTML_TAG "Une balise est manquante ou corrompue dans le fichier." << ENDL;
237  }
238  else
239  cerr << WARNING_HTML_TAG "Ligne n°" << lineN << ENDL;
240  }
241  }
242 
243  /* Tableau_entete va être en shared memory concurrent read access (no lock here) */
244 
245  unsigned int l = 0;
246 
247  for (const xmlChar* u : info.Table[info.NCumAgentXml])
248  {
249  if (l >= sizeof (Tableau_entete) / sizeof (char*)) break;
250 
251  ++l;
252 
253  if (u)
254  cerr << WARNING_HTML_TAG "Balise de paye : " << Tableau_entete[l]
255  << " " << u << ENDL;
256  }
257 
258  errorLine_t s = {lineN, string (info.threads->argv[info.fichier_courant]),
259  string ("Fichier : ") + string (info.threads->argv[info.fichier_courant])
260  + string (" -- Balise : ") + ((cur) ? string ((const char*)cur->name) : string ("NA"))
261  };
262  return s;
263 }

323 {
324 #if defined(__WIN32__)
325  /* Windows -------------------------------------------------- */
326  PROCESS_MEMORY_COUNTERS info;
327  GetProcessMemoryInfo ( GetCurrentProcess( ), &info, sizeof (info) );
328  return (size_t)info.WorkingSetSize;
329 
330 
331 #elif defined(__linux__) || defined(__linux) || defined(linux) || defined(__gnu_linux__)
332  /* Linux ---------------------------------------------------- */
333  long rss = 0L;
334  FILE* fp = NULL;
335 
336  if ( (fp = fopen ( "/proc/self/statm", "r" )) == NULL )
337  return (size_t)0L; /* Can't open? */
338 
339  if ( fscanf ( fp, "%*s%ld", &rss ) != 1 )
340  {
341  fclose ( fp );
342  return (size_t)0L; /* Can't read? */
343  }
344 
345  fclose ( fp );
346  return (size_t)rss * (size_t)sysconf ( _SC_PAGESIZE);
347 
348 #endif
349 }
