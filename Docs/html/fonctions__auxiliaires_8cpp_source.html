215 errorLine_t afficher_environnement_xhl (const info_t& info, const xmlNodePtr cur)
216 {
217 
218  long lineN = 0;
219  LOCK_GUARD
220  cerr << WARNING_HTML_TAG "Fichier analysé " << info.threads->argv[info.fichier_courant] << ENDL;
221 
222  if (cur)
223  {
224  lineN = (long) xmlGetLineNo (cur);
225 
226  if (lineN == 65535 && verbeux
227  && info.ligne_debut.size() > info.NCumAgentXml
228  && info.ligne_fin.size() > info.NCumAgentXml)
229  {
230  cerr << WARNING_HTML_TAG "Entre les lignes "
231  << info.ligne_debut.at (info.NCumAgentXml)[0] + 1
232  << " et "
233  << info.ligne_fin.at (info.NCumAgentXml)[0] << ENDL;
234  }
235  else
236  {
237  if (lineN == -1)
238  {
239  cerr << WARNING_HTML_TAG "Une balise est manquante ou corrompue dans le fichier." << ENDL;
240  }
241  else
242  cerr << WARNING_HTML_TAG "Ligne n°" << lineN << ENDL;
243  }
244  }
245 
246  /* Tableau_entete va être en shared memory concurrent read access (no lock here) */
247 
248  unsigned int l = 0;
249 
250  for (const xmlChar* u : info.Table[info.NCumAgentXml])
251  {
252  if (l >= sizeof (Tableau_entete) / sizeof (char*)) break;
253 
254  ++l;
255 
256  if (u)
257  cerr << WARNING_HTML_TAG "Balise de paye : " << Tableau_entete[l]
258  << " " << u << ENDL;
259  }
260 
261  errorLine_t s = {lineN, string (info.threads->argv[info.fichier_courant]),
262  string ("Fichier : ") + string (info.threads->argv[info.fichier_courant])
263  + string (" -- Balise : ") + ((cur) ? string ((const char*)cur->name) : string ("NA"))
264  };
265  return s;
266 }

325 size_t getCurrentRSS( )
326 {
327 #if defined(__WIN32__)
328  /* Windows -------------------------------------------------- */
329  PROCESS_MEMORY_COUNTERS info;
330  GetProcessMemoryInfo ( GetCurrentProcess( ), &info, sizeof (info) );
331  return (size_t)info.WorkingSetSize;
332 
333 
334 #elif defined(__linux__) || defined(__linux) || defined(linux) || defined(__gnu_linux__)
335  /* Linux ---------------------------------------------------- */
336  long rss = 0L;
337  FILE* fp = NULL;
338 
339  if ( (fp = fopen ( "/proc/self/statm", "r" )) == NULL )
340  return (size_t)0L; /* Can't open? */
341 
342  if ( fscanf ( fp, "%*s%ld", &rss ) != 1 )
343  {
344  fclose ( fp );
345  return (size_t)0L; /* Can't read? */
346  }
347 
348  fclose ( fp );
349  return (size_t)rss * (size_t)sysconf ( _SC_PAGESIZE);
350 
351 #endif
352 }
