354 inline xmlNodePtr GCC_INLINE atteindreNoeud(const char * noeud, xmlNodePtr cur)
355 {
356 
357 # ifdef DEBUG_ATTEINDRE
358 
359  cerr << "[DEBUG] --- Recherche de " << noeud << ENDL;
360 # endif
361 
362  while (cur && xmlIsBlankNode(cur))
363  {
364  cur = cur -> next;
365 # ifdef DEBUG_ATTEINDRE
366  cerr << "[DEBUG] Saut de noeud blanc" << ENDL;
367 # endif
368  }
369 
370 
371  while (cur != nullptr && xmlStrcmp(cur->name, (const xmlChar*) noeud))
372  {
373 # ifdef DEBUG_ATTEINDRE
374  // cerr << "[DEBUG] ......" << cur->name << ENDL;
375 # endif
376 
377  cur = cur->next;
378  }
379 
380  if (cur == nullptr)
381  {
382  AFFICHER_NOEUD(noeud) // cur->name == noeud
383  }
384 # ifdef DEBUG_ATTEINDRE
385  else
386  {
387  cerr << "[DEBUG] !!! Trouvé " << cur->name << ENDL;
388  }
389 # endif
390 
391 
392  return cur; // soit un pointer vers le bon noeud, soit nullptr
393 }
