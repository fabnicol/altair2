296 inline xmlNodePtr GCC_INLINE atteindreNoeud (const char * noeud, xmlNodePtr cur)
297 {
298 
299 # ifdef DEBUG_ATTEINDRE
300 
301  cerr << "[DEBUG] --- Recherche de " << noeud << ENDL;
302 # endif
303 
304  while (cur && xmlIsBlankNode (cur))
305  {
306  cur = cur -> next;
307 # ifdef DEBUG_ATTEINDRE
308  cerr << "[DEBUG] Saut de noeud blanc" << ENDL;
309 # endif
310  }
311 
312 
313  while (cur != nullptr && xmlStrcmp (cur->name, (const xmlChar*) noeud))
314  {
315 # ifdef DEBUG_ATTEINDRE
316  // cerr << "[DEBUG] ......" << cur->name << ENDL;
317 # endif
318 
319  cur = cur->next;
320  }
321 
322  if (cur == nullptr)
323  {
324  AFFICHER_NOEUD (noeud) // cur->name == noeud
325  }
326 
327 # ifdef DEBUG_ATTEINDRE
328  else
329  {
330  cerr << "[DEBUG] !!! Trouvé " << cur->name << ENDL;
331  }
332 
333 # endif
334 
335 
336  return cur; // soit un pointer vers le bon noeud, soit nullptr
337 }
