298 static inline xmlNodePtr GCC_INLINE atteindreNoeud (const char * noeud, xmlNodePtr cur)
299 {
300 
301 # ifdef DEBUG_ATTEINDRE
302 
303  cerr << "[DEBUG] --- Recherche de " << noeud << ENDL;
304 # endif
305 
306  while (cur && xmlIsBlankNode (cur))
307  {
308  cur = cur -> next;
309 # ifdef DEBUG_ATTEINDRE
310  cerr << "[DEBUG] Saut de noeud blanc" << ENDL;
311 # endif
312  }
313 
314 
315  while (cur != nullptr && xmlStrcmp (cur->name, (const xmlChar*) noeud))
316  {
317 # ifdef DEBUG_ATTEINDRE
318  // cerr << "[DEBUG] ......" << cur->name << ENDL;
319 # endif
320 
321  cur = cur->next;
322  }
323 
324  if (cur == nullptr)
325  {
326  AFFICHER_NOEUD (noeud) // cur->name == noeud
327  }
328 
329 # ifdef DEBUG_ATTEINDRE
330  else
331  {
332  cerr << "[DEBUG] !!! Trouvé " << cur->name << ENDL;
333  }
334 
335 # endif
336 
337 
338  return cur; // soit un pointer vers le bon noeud, soit nullptr
339 }
