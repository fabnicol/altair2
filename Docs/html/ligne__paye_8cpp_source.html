104 static void GCC_INLINE sanitize (xmlChar* s, const char sep)
105 {
106 
107  while (*s != 0)
108  {
109  // Non-switchable car info.seperateur n'est pas une expression constante.
110  if (*s == sep) *s = '_';
111 
112  switch (*s)
113  {
114 
115  case '\n':
116  *s = ' ';
117  break;
118 
119 #ifdef CONVERTIR_LATIN_1
120 #if !defined(__linux__) && ! defined(USE_ICONV)
121 
122 // Gros hack de pseudo-conversion UTF-8 vers Latin-1, qui permet d'économiser les 40 %
123 // de surcoût d'exécution lié à l'utilisation d'iconv pour retraiter les fichiers de
124 // sortie (fonction convertir(const char*))
125 // Ce hack est presque sans coût. Il se base sur les hypothèses suivantes :
126 // a) pas de caractères spéciaux multioctets
127 // b) seuls sont convertis : à, â, ç, è, é, ê, ë, î, ï, ô, û ... et les majuscules
128 // correspondantes càd
129 // dont le code UTF-8 commence par 0xC3. Il suffit d'ajouter 0x40 sur les quatre bits
130 // hauts de l'octet.
131 
132  case 0xC3:
133 
134  *s = ((* (s + 1) & 0xF0) + 0x40) | (* (s + 1) & 0x0F);
135 
136  effacer_char (s + 1);
137 
138  break;
139 
140  case 0xC2:
141 
142  *s = * (s + 1);
143 // Le caractère ° (degré) est bien codé en Latin-1 comme 0xB0, mais il y a un
144 // problème avec le paquet texlive inputenc pour la conversion pdf.
145 // On remplace donc par (0x65). Apparemment plus nécessaire */
146 // if (info.Table[info.NCumAgentXml][l][i] == 0xB0)
147 // info.Table[info.NCumAgentXml][l][i] = 0x65;
148 
149  effacer_char (s + 1);
150 
151  break;
152 #endif
153 #endif
154  }
155 
156  ++s;
157  }
158 }

174 static inline int GCC_INLINE Bulletin (const char* tag, xmlNodePtr& cur, int l, info_t& info)
175 {

177

179 
180  if ( nullptr == nextcur)
181  {
182  return NODE_NOT_FOUND;
183  }
184 
185 // On a à présent la garantie que cur->name correspond à tag
186 
187

191

678 static inline LineCount lignePaye (xmlNodePtr cur, info_t& info)
679 {
680  int l = BESOIN_MEMOIRE_ENTETE;
681 
682  int nbLignePaye = 0;
683 
684  unsigned int t = 0;
685 
686 // +1 pour éviter la confusion avec \0 des chaines vides
687  info.Table[info.NCumAgentXml][l] = (xmlChar*) xmlStrdup (drapeau[t]);
688  ++l;
689 
690 // Besoins en mémoire :
691 // BESOIN_MEMOIRE_ENTETE [champs hors ligne] + nombre de lignes
692 // + flags (maximum nbType * nb de rembobinages)
693 
694  int type_loop_counter = 0;
695 
696  while (cur != nullptr)
697  {
698  bool new_type = false;
699 
700  while (xmlStrcmp (cur->name, (const xmlChar *) type_remuneration[t]))
701  {
702  // Cas rare dans lequel <Remuneration> n'existe pas
703  if (xmlStrcmp (cur->name, (const xmlChar *) "NbHeureTotal") == 0)
704  return { nbLignePaye, l};
705 
706  ++t;
707 
708  if (t == nbType)
709  {
710 // En principe les éléments constitutifs des enregistrements
711 // <Remunération>....</Remuneration> sont enregistrés
712 // dans l'ordre du tableau type_remuneration. Toutefois quelques cas de
713 // désordre sont observés. Dans ces cas là on peut
714 // "rembobiner le tableau". On évite toutefois de faire une recherche
715 // ensembliste systématique, qui éviterait cela mais
716 // freinerait 99,9 % des recherches
717 
718  if (++type_loop_counter < TYPE_LOOP_LIMIT)
719  {
720  t = 0;
721  continue;
722  }
723 
724 // On ne rembobine qu'au maximum TYPE_LOOP_LIMIT. Si l'essai échoue,
725 // on déclenche une exception ou on retourne
726 
727  if (type_loop_counter == TYPE_LOOP_LIMIT)
728  {
729  cerr << ERROR_HTML_TAG "En excès du nombre de réordonnancements autorisés (" << type_loop_counter << "/" << TYPE_LOOP_LIMIT << ")." ENDL;
730  cerr << ERROR_HTML_TAG "En excès du nombre de types de lignes de paye autorisé (" << nbType << ")." ENDL;
731  }
732 
733  if (cur)
734  cerr << ERROR_HTML_TAG "Type litigieux " << cur->name
735  << " aux alentours du matricule "
736  << info.Table[info.NCumAgentXml][Matricule] << ENDL;
737  else
738  cerr << ERROR_HTML_TAG "Pointeur noeud courant nul" << ENDL;
739 
740 #ifdef STRICT
741  exit (-11);
742 #else
743  cerr << ERROR_HTML_TAG "Arrêt du décodage de la ligne de paye." << ENDL;
744  return {nbLignePaye, l};
745 #endif
746  }
747 
748  new_type = true;
749  }
750 
751  if (new_type && t < nbType)
752  {
753 // +1 pour éviter la confusion avec \0 des chaines vides
754  if ((info.Table[info.NCumAgentXml][l]
755  = (xmlChar*) xmlStrdup (drapeau[t])) == nullptr)
756  {
757  LOCK_GUARD
758 
759  if (verbeux) cerr << ERROR_HTML_TAG "Erreur dans l'allocation des drapeaux de catégories." << ENDL;
760 
761 #ifdef STRICT
762  exit (-12);
763 #else
764 
765  if (verbeux) cerr << ERROR_HTML_TAG "Arrêt du décodage de la ligne de paye." << ENDL;
766 
767 #endif
768  return {nbLignePaye, l};
769  }
770 
771  ++l;
772  }
773 
774 // ici on pourrait in fine se passer de ce test par compilation séparée
775 
776  if (! info.reduire_consommation_memoire)
777  {
778  verifier_taille (nbLignePaye, info);
779  }
780 
781 // Si on arrive à un noeud de type Commentaire, on le saute et on réinitialise
782 // "gratuitement" le parcours des drapeaux.
783 
784  if (xmlStrcmp (cur->name, (const xmlChar*) "Commentaire") == 0)
785  {
786  cur = cur->next;
787  t = 0;
788  --type_loop_counter; // 'Rembobinage gratuit'
789 
790  continue; // garantit incidemment que cur != nullptr dans la boucle
791  }
792 
793 // cur n'est pas nul à ce point
794 
795  cur = cur->xmlChildrenNode;
796 
797  if (cur == nullptr)
798  {
799  NA_ASSIGN (l);
800 #if 0
801 
802  // Normalement c'est ce qu'il faudrait faire
803  // Mais il n'y a pas eu de préallocation de mémoire pour ce cas
804 
805  for (short c = 0; c < 6; ++c) NA_ASSIGN (l++);
806 
807  ++nbLignePaye;
808 
809  if (verbeux) cerr << WARNING_HTML_TAG
810  "Anomalie : la ligne de paye est vide."
811  << ENDL;
812 
813 #endif
814  break;
815  }
816 
817 // cur n'est pas nul à ce point et ne devient jamais nul ci-après
818 
819 // Libellé, obligatoire
820 
821  bulletin_obligatoire ("Libelle", cur, l, info);
822 
823  ++l;
824 
825 // Code, obligatoire
826 
827  bulletin_obligatoire ("Code", cur, l, info);
828 
829  ++l;
830 
831 // Base, si elle existe
832 
833  bulletin_optionnel_numerique ("Base", cur, l, info);
834 
835  ++l;
836 
837 // Taux, s'il existe
838 
839  bulletin_optionnel_numerique ("Taux", cur, l, info);
840 
841  ++l;
842 
843 // Nombre d'unités, s'il existe
844 
845  bulletin_optionnel_numerique ("NbUnite", cur, l, info);
846 
847  ++l;
848 
849 // Montant obligatoire
850 
851  bulletin_obligatoire_numerique ("Mt", cur, l, info);
852 
853  ++l;
854 
855  if (t == 8) // La catégorie de ligne de paye est "Rappel"
856  {
857  if (cur->xmlChildrenNode == nullptr)
858  {
859  if (verbeux)
860  {
861  LOCK_GUARD
862  cerr << WARNING_HTML_TAG "Pas de période de référence pour le rappel" " pour le matricule " << info.Table[info.NCumAgentXml][Matricule] << " -- Ligne";
863  long lineN = xmlGetLineNo (cur);
864 
865  if (lineN != 65535)
866  {
867  cerr << " " << lineN << ENDL;
868  }
869  else
870  {
871  if (info.ligne_debut.size() > info.NCumAgentXml
872  && info.ligne_fin.size() > info.NCumAgentXml)
873  {
874  cerr << "s " << info.ligne_debut.at (info.NCumAgentXml)[0] + 1
875  << " - " << info.ligne_fin.at (info.NCumAgentXml)[0] + 1 << ENDL;
876  }
877  }
878  }
879 
880  NA_ASSIGN (l);
881  NA_ASSIGN (++l);
882  }
883  else
884  {
885  cur = cur->xmlChildrenNode;
886  // On ne tient pas rigueur du manque de qualité éventuelle
887  // tellement la norme est peu respectée
888 
889  bulletin_optionnel_char ("DateDebut", cur, l, info);
890 
891  ++l;
892  info.drapeau_cont = false; // pas de noeud successeur
893  bulletin_optionnel_char ("DateFin", cur, l, info);
894  cur = cur->parent->next;
895  info.drapeau_cont = true; // noeud successeur existe
896  }
897  }
898  else
899  {
900  NA_ASSIGN (l);
901  NA_ASSIGN (++l);
902  }
903 
904  ++l;
905 // cur ne sera pas nul à ce point
906 
907  ++nbLignePaye;
908 
909  cur = cur->parent->next;
910 
911 // Le parent ne peut être nul
912 // attention si du code est rajouté ici il doit l'être sous garde cur != nullptr
913 // Lorsque on a épuisé tous les types licites on a nécessairement cur = nullptr et
914 // la boucle s'arrête
915 
916  }
917 
918 
919  return { nbLignePaye, l};
920 }
