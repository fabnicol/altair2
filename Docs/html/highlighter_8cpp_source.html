98 void Highlighter::highlightBlock (const QString &text)
99 {
100  for (const HighlightingRule &rule : highlightingRules)
101  {
102  QRegExp expression (rule.pattern);
103  int index = expression.indexIn (text);
104 
105  while (index >= 0)
106  {
107  int length = expression.matchedLength();
108  setFormat (index, length, rule.format);
109  index = expression.indexIn (text, index + length);
110  }
111  }
