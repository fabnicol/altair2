164  return "codes." + X + " *<- *NA";

167 inline QString rempl_str (const QString &X, const QString &Y)
168 {
169  QStringList L = Y.split (";", QString::SkipEmptyParts);
170  QString Z;
171 
172  if (L.size() >= 0)
173  {
174  for (QString& s : L)
175  {
176  s = "\"" + s + "\"";
177  }
178 
179  Z = "c(" + L.join (",") + ")";
180  }
181  else
182  return "codes." + X + " <- NA";
183 
184  return "codes." + X + " <- " + Z;
185 }

217 void codePage::substituer_valeurs_dans_script_R()
218 {
219  reinitialiser_prologue();
220  QString file_str = common::readFile (prologue_codes_path);
221  common::exporter_identification_controle (file_str);
222  bool res = false;
223  bool res2 = true;
224 
225  QIcon icon0 = QIcon (":/images/view-refresh.png");
226  QIcon icon1 = QIcon (":/images/msg.png");
227  QIcon icon2 = QIcon (":/images/error.png");
228 
229  QIcon icon = (appliquerCodes->isChecked()) ?
230  icon1 :
231  icon0;
232 
233  for (const FLineEdit* a : listeCodes)
234  {
235  bool test = ! a->text().isEmpty();
236  res |= test;
237  res2 &= test;
238  }
239 
240  if (res == false)
241  {
242  Warning ("Attention",
243  "Les codes sont tous non renseignés.<br>"
244  "Les tests statutaires se feront <br>"
245  "sous algorithme heuristique seulement.");
246 
247  icon = icon2;
248 
249  res = renommer (dump (file_str), prologue_codes_path);
250 
251  if (Hash::Reference.isEmpty()) return;
252 
253  if (res == true)
254  label->setText ("L'identification du contrôle a été exportée dans le rapport.");
255  else
256  label->setText ("Erreur d'enregistrement du fichier de configuration prologue_codes.R");
257 
258  return;
259  }
260  else if (res2 == false)
261  {
262  QString liste_codes_nr;
263  int i = 0;
264 
265  for (const FLineEdit* a : listeCodes)
266  {
267  if (a->text().isEmpty())
268  liste_codes_nr += listeLabels[i].toUpper() + "<br>";
269 
270  ++i;
271  }
272 
273  Warning ("Attention",
274  "Certains code ne sont pas renseignés.<br>"
275  "Les tests statutaires se feront <br>"
276  "sous algorithme heuristique pour :<br>"
277  +
278  liste_codes_nr);
279  }
280 
281  QString liste_codes;
282 
283  for (int rang = 0; rang < listeCodes.size(); ++rang)
284  {
285  const QString &s = listeLabels.at (rang);
286  const QString &codes = listeCodes.at (rang)->text();
287  res = true;
288 
289  if (! codes.isEmpty())
290  {
291  res = substituer (regexp (s), rempl_str (s, codes), file_str);
292  liste_codes += "<li>" + listeLabels.at (rang).toUpper() + " : " + codes + "</li>";
293  }
294 
295  if (res == false)
296  {
297  Warning ("Attention",
298  "Le remplacement de la variable codes." + s + "<br>"
299  "n'a pas pu être effectué dans le fichier prologue_codes.R<br>"
300  "Les tests statutaires se feront<br>"
301  "sous algorithme heuristique seulement.");
302 
303  icon = icon2;
304  }
305  }
306 
307  res = renommer (dump (file_str), prologue_codes_path);
308 
309  appliquerCodes->setIcon (icon);
310 
311  if (res == true)
312  label->setText ("Les codes de paye suivants :"
313  "<ul>" + liste_codes + "</ul>"
314  "seront pris en compte pour les rapports.");
315  else
316  label->setText ("Erreur d'enregistrement du fichier de configuration prologue_codes.R");
317 }
