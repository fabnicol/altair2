166 inline QString rempl_str (const QString &X, const QString &Y)
167 {
168  QStringList L = Y.split (";", QString::SkipEmptyParts);
169  QString Z;
170 
171  if (L.size() >= 0)
172  {
173  for (QString& s : L)
174  {
175  s = "\"" + s + "\"";
176  }
177 
178  Z = "c(" + L.join (",") + ")";
179  }
180  else
181  return "codes." + X + " <- NA";
182 
183  return "codes." + X + " <- " + Z;
184 }
