410 inline const QString Altair::makeParserString (int start, int end)
411 {
412  QStringList L = QStringList();
413  auto origin = Abstract::abstractWidgetList.begin();
414 
415  // Parcourir l'ensemble de la liste abstractWidgetList des fwidgets
416 
417  for (auto u = origin + start; u != Abstract::abstractWidgetList.end() && u <= origin + end; ++u)
418  {
419  FAbstractWidget* v = *u;
420 
421  if (v == nullptr) return "";
422 
423  QString hK = v->getHashKey();
424 
425  if (hK.isEmpty())
426  {
427  Warning (tr ("Erreur"), tr ("Erreur d'analyse XML du projet"));
428  continue;
429  }
430 
431  // Enregistrer le projet XML à partir du contenu des widgets
432 
433  QString xml = v->setXmlFromWidget().toQString();
434 
435  if (hK == "XHL" && xml.isEmpty()) continue;
436 
437  // La profondeur permet d'ecrire des objets complexes
438 
439  QString widgetDepth = v->getDepth();
440 
441  L << " <" + hK + " profondeur=\"" + widgetDepth + "\">\n "
442  + xml
443  + "\n </" + hK + ">\n";
444  }
445 
446  // Retourner le QString du projet .alt
447 
448  return L.join ("");
449 }

714 inline qint64 displaySecondLevelData (const QStringList &tags,
715  const QVector<QStringList> &stackedInfo,
716  const QVector<QStringList> &stackedSizeInfo)
717 {
718  int count = 0, tagcount = 0, l;
719  qint64 filesizecount = 0;
720 
721  QString firstColumn,
722  root = tags.at (0),
723  secondColumn = " ",
724  thirdColumn,
725  fourthColumn,
726  fifthColumn,
727  sixthColumn,
728  seventhColumn;
729 
730  int tagListSize = tags.size();
731 
732  QVectorIterator<QStringList> i (stackedInfo), j (stackedSizeInfo);
733 
734  while (i.hasNext() && j.hasNext())
735  {
736  if (!root.isEmpty() && tagcount < tagListSize)
737  firstColumn = tags.at (tagcount++);
738 
739  if (firstColumn[0] != '2') break;
740 
741  displayTextData ({firstColumn});
742 
743  QStringListIterator w (i.next()), y (j.next());
744  l = 0;
745 
746  while (w.hasNext() && y.hasNext())
747  {
748  ++count;
749 
750  thirdColumn = "fichier " + QString::number (++l) + "/" + QString::number (count) + ": ";
751  const QString filename = w.next();
752  thirdColumn += filename;
753  secondColumn = Hash::Mois[filename];
754  sixthColumn = "";
755 
756  for (int s = 0; s < Hash::Siret[filename].size() && s < Hash::Etablissement[filename].size(); ++s)
757  {
758  sixthColumn += ((s > 0) ? "\n" : "") + Hash::Employeur[filename];
759  sixthColumn += " " + Hash::Siret[filename].at (s);
760  sixthColumn += " " + Hash::Etablissement[filename].at (s);
761  }
762 
763  seventhColumn = Hash::Budget[filename];
764 
765  if ((stackedSizeInfo.size() > 0) && (y.hasNext()))
766  {
767  QStringList units = y.next().split (" ");
768  qint64 msize = units.at (0).toLongLong();
769  filesizecount += msize;
770  // force coertion into float or double using .0
771  fourthColumn = QString::number (msize / 1048576.0, 'f', 1);
772  fifthColumn = QString::number (filesizecount / 1048576.0, 'f', 1) + " Mo" ;
773  }
774 
775  displayTextData ({""},
776  secondColumn,
777  thirdColumn,
778  fourthColumn,
779  fifthColumn,
780  sixthColumn,
781  seventhColumn,
782  (y.hasNext()) ? QColor ("navy") : ((j.hasNext()) ? QColor ("orange") : QColor ("red")));
783  }
784  }
785 
786  return filesizecount;
787 }
