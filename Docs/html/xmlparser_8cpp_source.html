51 inline const QString Altair::makeParserString (int start, int end)
52 {
53  QStringList L = QStringList();
54  auto origin = Abstract::abstractWidgetList.begin();
55 
56  // Parcourir l'ensemble de la liste abstractWidgetList des fwidgets
57 
58  for (auto u = origin + start; u != Abstract::abstractWidgetList.end() && u <= origin + end; ++u)
59  {
60  FAbstractWidget* v = *u;
61 
62  if (v == nullptr) return "";
63 
64  QString hK = v->getHashKey();
65 
66  if (hK.isEmpty())
67  {
68  Warning (tr ("Erreur"), tr ("Erreur d'analyse XML du projet"));
69  continue;
70  }
71 
72  // Enregistrer le projet XML à partir du contenu des widgets
73 
74  QString xml = v->setXmlFromWidget().toQString();
75 
76  if (hK == "XHL" && xml.isEmpty()) continue;
77 
78  // La profondeur permet d'ecrire des objets complexes
79 
80  QString widgetDepth = v->getDepth();
81 
82  L << " <" + hK + " profondeur=\"" + widgetDepth + "\">\n "
83  + xml
84  + "\n </" + hK + ">\n";
85  }
86 
87  // Retourner le QString du projet .alt
88 
89  return L.join ("");
90 }

355 inline qint64 displaySecondLevelData (const QStringList &tags,
356  const QVector<QStringList> &stackedInfo,
357  const QVector<QStringList> &stackedSizeInfo)
358 {
359  int count = 0, tagcount = 0;
360  qint64 filesizecount = 0;
361 
362  QString firstColumn,
363  root = tags.at (0),
364  secondColumn = " ",
365  thirdColumn,
366  fourthColumn,
367  fifthColumn,
368  sixthColumn,
369  seventhColumn;
370 
371  int tagListSize = tags.size();
372 
373  QVectorIterator<QStringList> i (stackedInfo), j (stackedSizeInfo);
374 
375  while (i.hasNext() && j.hasNext())
376  {
377  if (!root.isEmpty() && tagcount < tagListSize)
378  firstColumn = tags.at (tagcount++);
379 
380  if (firstColumn[0] != '2') break;
381 
382  displayTextData ({firstColumn});
383 
384  QStringListIterator w (i.next()), y (j.next());
385 
386  int l = 0;
387 
388  while (w.hasNext() && y.hasNext())
389  {
390  ++count;
391 
392  thirdColumn = "fichier " + QString::number (++l) + "/" + QString::number (count) + ": ";
393  const QString filename = w.next();
394  thirdColumn += filename;
395  secondColumn = Hash::Mois[filename];
396  sixthColumn = "";
397 
398  for (int s = 0; s < Hash::Siret[filename].size() && s < Hash::Etablissement[filename].size(); ++s)
399  {
400  sixthColumn += ((s > 0) ? "\n" : "") + Hash::Employeur[filename];
401  sixthColumn += " " + Hash::Siret[filename].at (s);
402  sixthColumn += " " + Hash::Etablissement[filename].at (s);
403  }
404 
405  seventhColumn = Hash::Budget[filename];
406 
407  if ((stackedSizeInfo.size() > 0) && (y.hasNext()))
408  {
409  QStringList units = y.next().split (" ");
410  qint64 msize = units.at (0).toLongLong();
411  filesizecount += msize;
412  // force coertion into float or double using .0
413  fourthColumn = QString::number (msize / 1048576.0, 'f', 1);
414  fifthColumn = QString::number (filesizecount / 1048576.0, 'f', 1) + " Mo" ;
415  }
416 
417  displayTextData ({""},
418  secondColumn,
419  thirdColumn,
420  fourthColumn,
421  fifthColumn,
422  sixthColumn,
423  seventhColumn,
424  (y.hasNext()) ? QColor ("navy") : ((j.hasNext()) ? QColor ("orange") : QColor ("red")));
425  }
426  }
427 
428  return filesizecount;
429 }

452 void Altair::parseProjectFile (QIODevice* file)
453 {

507 
508  // Empiler les informations des différentes balises
509  // dans une FStringList et ajouter à la table de hashage Hash::wrapper
510  // dont la clé est la première balise

517  subnode = subnode.nextSibling();
