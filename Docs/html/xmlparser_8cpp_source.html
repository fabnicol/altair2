233 inline const QString Altair::makeParserString (int start, int end)
234 {
235  QStringList L = QStringList();
236  auto origin = Abstract::abstractWidgetList.begin();
237 
238  // Parcourir l'ensemble de la liste abstractWidgetList des fwidgets
239 
240  for (auto u = origin + start; u != Abstract::abstractWidgetList.end() && u <= origin + end; ++u)
241  {
242  FAbstractWidget* v = *u;
243 
244  if (v == nullptr) return "";
245 
246  QString hK = v->getHashKey();
247 
248  if (hK.isEmpty())
249  {
250  Warning (tr ("Erreur"), tr ("Erreur d'analyse XML du projet"));
251  continue;
252  }
253 
254  // Enregistrer le projet XML à partir du contenu des widgets
255 
256  QString xml = v->setXmlFromWidget().toQString();
257 
258  if (hK == "XHL" && xml.isEmpty()) continue;
259 
260  // La profondeur permet d'ecrire des objets complexes
261 
262  QString widgetDepth = v->getDepth();
263 
264  L << " <" + hK + " profondeur=\"" + widgetDepth + "\">\n "
265  + xml
266  + "\n </" + hK + ">\n";
267  }
268 
269  // Retourner le QString du projet .alt
270 
271  return L.join ("");
272 }

537 inline qint64 displaySecondLevelData (const QStringList &tags,
538  const QVector<QStringList> &stackedInfo,
539  const QVector<QStringList> &stackedSizeInfo)
540 {
541  int count = 0, tagcount = 0, l;
542  qint64 filesizecount = 0;
543 
544  QString firstColumn,
545  root = tags.at (0),
546  secondColumn = " ",
547  thirdColumn,
548  fourthColumn,
549  fifthColumn,
550  sixthColumn,
551  seventhColumn;
552 
553  int tagListSize = tags.size();
554 
555  QVectorIterator<QStringList> i (stackedInfo), j (stackedSizeInfo);
556 
557  while (i.hasNext() && j.hasNext())
558  {
559  if (!root.isEmpty() && tagcount < tagListSize)
560  firstColumn = tags.at (tagcount++);
561 
562  if (firstColumn[0] != '2') break;
563 
564  displayTextData ({firstColumn});
565 
566  QStringListIterator w (i.next()), y (j.next());
567  l = 0;
568 
569  while (w.hasNext() && y.hasNext())
570  {
571  ++count;
572 
573  thirdColumn = "fichier " + QString::number (++l) + "/" + QString::number (count) + ": ";
574  const QString filename = w.next();
575  thirdColumn += filename;
576  secondColumn = Hash::Mois[filename];
577  sixthColumn = "";
578 
579  for (int s = 0; s < Hash::Siret[filename].size() && s < Hash::Etablissement[filename].size(); ++s)
580  {
581  sixthColumn += ((s > 0) ? "\n" : "") + Hash::Employeur[filename];
582  sixthColumn += " " + Hash::Siret[filename].at (s);
583  sixthColumn += " " + Hash::Etablissement[filename].at (s);
584  }
585 
586  seventhColumn = Hash::Budget[filename];
587 
588  if ((stackedSizeInfo.size() > 0) && (y.hasNext()))
589  {
590  QStringList units = y.next().split (" ");
591  qint64 msize = units.at (0).toLongLong();
592  filesizecount += msize;
593  // force coertion into float or double using .0
594  fourthColumn = QString::number (msize / 1048576.0, 'f', 1);
595  fifthColumn = QString::number (filesizecount / 1048576.0, 'f', 1) + " Mo" ;
596  }
597 
598  displayTextData ({""},
599  secondColumn,
600  thirdColumn,
601  fourthColumn,
602  fifthColumn,
603  sixthColumn,
604  seventhColumn,
605  (y.hasNext()) ? QColor ("navy") : ((j.hasNext()) ? QColor ("orange") : QColor ("red")));
606  }
607  }
608 
609  return filesizecount;
610 }

633 void Altair::parseProjectFile (QIODevice* file)
634 {

636  // pass it as a parameter to the constructor otherwise the protected parent member will be accessible yet unaltered
