585 pair<uint64_t, uint32_t> boucle_ecriture(vector<info_t>& Info, int nsegment)
586 {
587  int ligne = 0;
588  uint64_t compteur = 0,
589  compteur_annee_courante = 0;
590  static uint64_t dernier_compteur;
591 
592  uint64_t compteur_lignes_bulletins = 0;
593 
594  char sep = Info[0].separateur;
595  char* annee_courante = (char*) Info[0].Table[0][Annee];
596  static unsigned rang_fichier_base = 1,rang_fichier_base_annee_courante;
597  static ofstream base;
598  static ofstream bulletins;
599  static array<ofstream, nbType + 1> tableau_base;
600 
601 #ifdef OFSTREAM_TABLE_OUTPUT // cas de l'écriture directe dans le fichier base
602 
603 #define t_base base
604 #define t_bulletins bulletins
605 #define t_tableau_base tableau_base
606 
607 #else // case de l'écriture dans un tampon string avant écriture sur le disque (nécessite plus de mémoire)
608 
609  ostringstream t_base;
610  ostringstream t_bulletins;
611  static array<ostringstream, nbType + 1> t_tableau_base;
612 
613 #endif
614 
615  ouvrir_fichier_bulletins(Info[0], bulletins, nsegment);
616 
617  uint32_t taille_base = Info[0].taille_base;
618  BaseType type_base = Info[0].type_base;
619 
620 
621  if (Info[0].generer_rang)
622  {
623  if (Info[0].select_echelon)
624  {
625  ecrire_ligne_table = ECRIRE_LIGNE_l_GENERER_RANG_ECHELON;
626  ecrire_ligne_bulletin = ECRIRE_LIGNE_BULLETIN_GENERER_RANG_ECHELON;
627  }
628  else
629  {
630  ecrire_ligne_table = ECRIRE_LIGNE_l_GENERER_RANG;
631  ecrire_ligne_bulletin = ECRIRE_LIGNE_BULLETIN_GENERER_RANG;
632  }
633  }
634  else
635  if (Info[0].select_siret)
636  {
637  if (Info[0].select_echelon)
638  {
639  ecrire_ligne_table = ECRIRE_LIGNE_l_SIRET_ECHELON;
640  ecrire_ligne_bulletin = ECRIRE_LIGNE_BULLETIN_SIRET_ECHELON;
641  }
642  else
643  {
644  ecrire_ligne_table = ECRIRE_LIGNE_l_SIRET;
645  ecrire_ligne_bulletin = ECRIRE_LIGNE_BULLETIN_SIRET;
646  }
647 
648  }
649  else
650  {
651  if (Info[0].select_echelon)
652  {
653  ecrire_ligne_table = ECRIRE_LIGNE_l_ECHELON;
654  ecrire_ligne_bulletin = ECRIRE_LIGNE_BULLETINS_ECHELON;
655  }
656  else
657  {
658  ecrire_ligne_table = ECRIRE_LIGNE_l;
659  ecrire_ligne_bulletin = ECRIRE_LIGNE_BULLETINS;
660  }
661  }
662 
663  switch (type_base)
664  {
665  case BaseType::TOUTES_CATEGORIES :
666  for (int d = 0; d <= nbType; ++d) // Il faut tenir compte des types NA
667  {
668  ouvrir_fichier_base(Info[0], static_cast<BaseType>(d + 1), tableau_base[d], nsegment);
669  if (! tableau_base[d].is_open())
670  {
671  cerr << ERROR_HTML_TAG "impossible d'ouvrir la base de lignes de paye." ENDL;
672  exit(2003);
673  }
674  }
675 
676  break;
677 
678  default : ouvrir_fichier_base(Info[0], type_base, base, nsegment);
679  if (! base.is_open())
680  {
681  cerr << ERROR_HTML_TAG "impossible d'ouvrir la base de lignes de paye." ENDL;
682  exit(2004);
683  return make_pair(0,0);
684  }
685 
686  }
687 
688  if (! bulletins.is_open())
689  {
690  cerr << ERROR_HTML_TAG "impossible d'ouvrir la base de bulletins de paye." ENDL;
691  return make_pair(0,0);
692  }
693 
694 #ifdef GENERATE_RANK_SIGNAL
695  uint64_t NCumLignes = 0;
696  int progression = 0;
697  unsigned int step = 0;
698 
699  for (unsigned i = 0; i < Info[0].nbfil; ++i)
700  {
701  NCumLignes += Info[i].nbLigne;
702  }
703 
704  reset_rank_signal();
705  generate_rank_signal();
706 
707 #endif
708 
709  static uint64_t compteur_ancien, compteur_bulletins_ancien, compteur_ancien_annee_courante;
710  compteur = compteur_ancien;
711  compteur_lignes_bulletins = compteur_bulletins_ancien;
712 
713  if (type_base == BaseType::MONOLITHIQUE)
714  {
715  for (unsigned i = 0; i < Info[0].nbfil; ++i)
716  {
717  for (uint32_t agent = 0; agent < Info[i].NCumAgentXml; ++agent)
718  {
719  unsigned l = BESOIN_MEMOIRE_ENTETE;
720  uint16_t NLigneAgent = Info[i].NLigne[agent];
721 
722  compteur_lignes_bulletins += ecrire_ligne_bulletin(i, agent, t_bulletins, sep, Info, compteur_lignes_bulletins);
723 
724  char* type = nullptr;
725 
726  while (ligne < NLigneAgent)
727  {
728  int test_drapeau_categorie;
729 
730  // teste si un drapeau de nouvelle catégorie de ligne de paye (T, I,...) a été introduit en base
731  if (VAR(l+1) && xmlStrcmp(VAR(l+1), NA_STRING) == 0)
732  {
733  type = (char*) NA_STRING;
734  }
735  else
736  while (VAR(l) && (test_drapeau_categorie = VAR(l)[0], test_drapeau_categorie <= nbType) && (test_drapeau_categorie >= 1))
737  {
738  type = const_cast<char*>(type_remuneration_traduit[test_drapeau_categorie - 1]);
739  ++l;
740  // no break !!!!
741  }
742 
743  compteur += ecrire_ligne_table(i, agent, l, type, t_base, sep, Info, compteur);
744 
745  l += INDEX_MAX_COLONNNES + 1;
746  ++ligne;
747  }
748 
749  ligne = 0;
750 
751 #ifdef GUI_TAG_MESSAGES
752 #ifdef GENERATE_RANK_SIGNAL
753  progression = ceil((float) (compteur * 100) / (float) NCumLignes );
754 
755  ++step;
756 
757  if (step > Info[i].NCumAgentXml / 5)
758  {
759  generate_rank_signal(progression);
760  cerr << " \n";
761  step = 0;
762  }
763 #endif
764 #endif
765  }
766  }
767  }
768  else
769  if (type_base == BaseType::PAR_ANNEE)
770  {
771  for (unsigned i = 0; i < Info[0].nbfil; ++i)
772  {
773  for (uint32_t agent = 0; agent < Info[i].NCumAgentXml; ++agent)
774  {
775  unsigned l = BESOIN_MEMOIRE_ENTETE;
776  uint16_t NLigneAgent = Info[i].NLigne[agent];
777 
778  compteur_lignes_bulletins += ecrire_ligne_bulletin(i, agent, t_bulletins, sep, Info, compteur_lignes_bulletins);
779 
780  if (strcmp((const char*)VAR(Annee), annee_courante))
781  {
782 #ifndef OFSTREAM_TABLE_OUTPUT // Il faut écrire dans le fichier OFSTREAM la chaine de caractères temporaires
783  base << t_base.str();
784  t_base.str("");
785 #endif
786 
787  base.close();
788 
789  cerr << "Année : " << annee_courante << ENDL;
790  cerr << "Table de " << compteur - dernier_compteur
791  << " lignes, lignes " << dernier_compteur + 1
792  << " à " << compteur << "." ENDL ENDL;
793 
794  dernier_compteur = compteur;
795  compteur_annee_courante = 0;
796 
797  annee_courante = (char*) VAR(Annee);
798  ouvrir_fichier_base(Info[i], type_base, base, nsegment);
799  if (! base.is_open()) return make_pair(0,0);
800  }
801 
802  char* type = nullptr;
803 
804  while (ligne < NLigneAgent)
805  {
806  int test_drapeau_categorie;
807 
808  // teste si un drapeau de nouvelle catégorie de ligne de paye (T, I,...) a été introduit en base
809  if (VAR(l+1) && xmlStrcmp(VAR(l+1), NA_STRING) == 0)
810  {
811  type = (char*) NA_STRING;
812  }
813  else
