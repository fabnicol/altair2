544 void Altair::deleteGroup()
545 {
546  updateIndexInfo();
547 
548  // Taille du conteneur de données
549  // project->widgetContainer
550  // moins 1, autrement dit du nombre d'onglets contenant des fichiers de paye, moins 1
551 
552  uint rank = (uint) project->getRank();
553 
554  // le conteneur fileSizeDataBase contient toutes les tailles de fichier pour tous les
555  // onglets. On vide le conteneur à l'index correspondant à l'onglet supprimé
556 
557  if ((uint) fileSizeDataBase[0].size() > currentIndex)
558  fileSizeDataBase[0][currentIndex].clear();
559 
560  // Il faut se fonder sur rank plutot que sur currentIndex, qui peut être sur des
561  // onglets supplémentaires (type Siret...) qui ne correspondent pas à des données
562  // de fichier
563 
564  if (rank > 0)
565  {
566  // Pour tous les onglets de rang supérieur à l'onglet courant et inférieur à l'onglet
567  // maximal contenant des données, translater les données de taille de fichier d'un
568  // index vers la gauche après suppression de l'onglet courant
569 
570  if (currentIndex < rank)
571  {
572 
573  for (unsigned j = currentIndex; j < rank ; j++)
574  {
575  fileSizeDataBase[0][j] = fileSizeDataBase[0][j + 1];
576  }
577  }
578  }
579 
580  // Réactualiser le projet .alt et le gestionnaire de projet à droite de l'interface
581 
582  updateProject();
583 
584  // Afficher la taille des bases de paye dans l'onglet Messages
585  // Elle sera en principe inférieure à la taille avant suppression de l'onglet
586 
587  displayTotalSize();
588 }

661 void Altair::assignWidgetValues()
662 {

664
