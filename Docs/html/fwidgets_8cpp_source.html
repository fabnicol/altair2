200 const QStringList FAbstractWidget::commandLineStringList()
201 {
202  /* If command line option is ill-formed, or if a corresponding checkbox is unchecked (or negatively checked)
203  * or if an argument-taking option has no-argument, return empty */
204 //Warning(nullptr, "", optionLabel+":"+QString::number(commandLineType & flags::::widgetMask));
205 
206  if (
207  optionLabel.isEmpty()
208  || commandLineList.isEmpty()
209  || commandLineList[0].isFalse()
210  // || (commandLineList[0].toQString().isEmpty())
211  || (this->isAbstractDisabled())) return {};
212 
213  QStringList strL=QStringList();
214 
215  if ((this->status & flags::status::widgetMask) == flags::status::hasListCommandLine)
216  {
217  QListIterator<FString> i(commandLineList);
218  while (i.hasNext())
219  if (!i.peekNext().isEmpty())
220  strL << i.next().trimmed();
221  else
222  i.next();
223 
224  }
225  else
226  {
227  if (commandLineList[0].isTrue() | commandLineList[0].isMultimodal())
228  {
229  if (optionLabel.size() == 1)
230  strL = QStringList("-"+optionLabel);
231  else
232  {
233  if (optionLabel.at(0) == '^')
234  strL = QStringList(optionLabel.mid(1).trimmed());
235  else
236  strL = QStringList ("--" +optionLabel);
237  }
238  }
239  else
240  {
241  if (! commandLineList[0].toQString().isEmpty())
242  {
243  if (optionLabel.at(0) == '^')
244  {
245  if (commandLineList[0] != " ")
246  strL = QStringList(optionLabel.mid(1).trimmed() + "=" + commandLineList[0].toQString());
247  }
248  else
249  if (optionLabel.at(0) == '%')
250  {
251  if (commandLineList[0] != " ")
252  strL = (optionLabel.size() == 1)? (QStringList() << "-" + optionLabel.mid(1) << commandLineList[0].toQString())
253  : (QStringList() << "--" + optionLabel.mid(1) << commandLineList[0].toQString());
254  }
255  else
256  {
257  if (commandLineList[0] != " ")
258  {
259  strL = (optionLabel.size() == 1)? (QStringList() << "-"+optionLabel << commandLineList[0].toQString())
260  :(QStringList() << "--" + optionLabel+"=" << commandLineList[0].toQString());
261  }
262  }
263  }
264  }
265  }
266 
267  return QStringList(strL);
268 }
