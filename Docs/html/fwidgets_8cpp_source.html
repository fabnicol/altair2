61 
62  QVectorIterator<QWidget*> componentlistIterator(w->getComponentList());

64  while ((componentlistIterator.hasNext()) && (objectlistIterator.hasNext()))

67  QVectorIterator<QWidget*> i(objectlistIterator.next());
68  while (i.hasNext())
69  {
70  QWidget* item=i.next();
71  if ((item == nullptr) || (component==nullptr)) continue;
72  // This does not always work automatically through Qt parenting as it normally should, so it is necessary to reimplement enabling dependencies
73  // e.g. for QLabels

76  item->setEnabled(false);
77  else
78  if (component->metaObject()->className() == QString("FCheckBox"))
79  item->setEnabled(static_cast<FCheckBox*>(component)->isChecked());
80 
81  connect(component, SIGNAL(toggled(bool)), item, SLOT(setEnabled(bool)));
82 
83  // this connection must follow the general enabling adjustments above
84  // A way to avoid this default behaviour is to group the enabled boxes into a flat (invisible) GroupBox
85 
86  if (item->metaObject()->className() == QString("FCheckBox"))
87  connect(component, SIGNAL(toggled(bool)), item, SLOT(uncheckDisabledBox()));
88  else
89  if (item->metaObject()->className() == QString("FRadioBox"))

92 
93  }
94  }
95  }
96 
97  if ((disabledObjects != nullptr) && (!disabledObjects->isEmpty()))
98  {
99 
100  QVectorIterator<QWidget*> newcomponentlistIterator(w->getComponentList());
101  Q2VectorIterator newobjectlistIterator(*disabledObjects);

199 const QStringList FAbstractWidget::commandLineStringList()
200 {
201  /* If command line option is ill-formed, or if a corresponding checkbox is unchecked (or negatively checked)
202  * or if an argument-taking option has no-argument, return empty */
203 //Warning(nullptr, "", optionLabel+":"+QString::number(commandLineType & flags::::widgetMask));
204 
205  if (
206  optionLabel.isEmpty()
207  || commandLineList.isEmpty()
208  || commandLineList[0].isFalse()
209  // || (commandLineList[0].toQString().isEmpty())
210  || (this->isAbstractDisabled())) return {};
211 
212  QStringList strL=QStringList();
213 
214  if ((this->status & flags::status::widgetMask) == flags::status::hasListCommandLine)
215  {
216  QListIterator<FString> i(commandLineList);
217  while (i.hasNext())
218  if (!i.peekNext().isEmpty())
219  strL << i.next().trimmed();
220  else
221  i.next();
222 
223  }
224  else
225  {
226  if (commandLineList[0].isTrue())
227  {
228  if (optionLabel.size() == 1)
229  strL = QStringList("-"+optionLabel);
230  else
231  {
232  if (optionLabel.at(0) == '^')
233  strL = QStringList(optionLabel.mid(1).trimmed());
234  else
235  strL = QStringList ("--" +optionLabel);
236  }
237  }
238  else
239  {
240  if (! commandLineList[0].toQString().isEmpty())
241  {
242  if (optionLabel.at(0) == '^')
243  {
244  if (commandLineList[0] != " ")
245  strL = QStringList(optionLabel.mid(1).trimmed() + "=" + commandLineList[0].toQString());
246  }
247  else
248  if (optionLabel.at(0) == '%')
249  {
250  if (commandLineList[0] != " ")
251  strL = (optionLabel.size() == 1)? (QStringList() << "-" + optionLabel.mid(1) << commandLineList[0].toQString())
252  : (QStringList() << "--" + optionLabel.mid(1) << commandLineList[0].toQString());
253  }
254  else
255  {
256  if (commandLineList[0] != " ")
257  {
258  strL = (optionLabel.size() == 1)? (QStringList() << "-"+optionLabel << commandLineList[0].toQString())
259  :(QStringList() << "--" + optionLabel+"=" << commandLineList[0].toQString());
260  }
261  }
262  }
263  }
264  }
265 
266  return QStringList(strL);
267 }
