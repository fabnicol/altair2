201 void FListFrame::launch_thread(int rank)
202 {
203  if (isTerminated || size == 0 || rank >= size) return;
204 
205  // Ajouter un fil d'exécution à la liste des fils thread
206 
207  thread.push_back(new QThread);
208 
209  const QString& fileName = stringList.at(rank);
210 
211  // Démarrer le fil et lui faire lire l'entête du fichier XHL en cours
212 
213  thread[rank]->start();
214  connect(thread[rank], &QThread::started, [this, fileName] {
215  parseXhlFile(fileName);
216  });
217 
218  // Lorsque le fil a fini de lire l'année, le mois etc. le signal parsed() est émis et le fil doit être arrêté

222  // Dès lors on peut passer au fil suivant et au fichier suivant

232 void FListFrame::parseXhlFile(const QString& fileName)
233 {
234 
235  QFile file(fileName);
236  long long ligne = 0;
237  bool result = file.open(QIODevice::ReadOnly);
238  if (! file.isOpen())
239  emit(textAppend(ERROR_HTML_TAG "Erreur à l'ouverture du fichier."));
240 
241  if (result == false || file.size()== 0)
242  {
243  emit(textAppend(WARNING_HTML_TAG "Fichier vide."));
244  return;
245  }
246 
247  file.seek(0);
248  QByteArray buffer0 = file.readAll();
249  QString string = QString::fromLatin1(buffer0, BUFFER_SIZE);
250 
251 # define QUOTE "(?:\"|')"
252 
253  string.remove(QRegularExpression("[\\(\\)]"));
254  QRegExp reg("DocumentPaye.*(?:Annee) V.?=.?" QUOTE "([0-9]+)" QUOTE ".*(?:Mois) V.?=.?" QUOTE "([0-9]+)" QUOTE "(.*)(?:Employeur).*(?:Nom) V.?=.?" QUOTE "([^" QUOTE "]+)" QUOTE ".*(?:Siret) V.?=.?" QUOTE "([0-9A-Z]+)" QUOTE ".*DonneesIndiv(.*)PayeIndivMensuel");
255  reg.setPatternSyntax(QRegExp::RegExp2);
256  reg.setCaseSensitivity(Qt::CaseInsensitive);
257  QRegExp reg2(".*Budget.*Libelle V.?=.?" QUOTE "([^" QUOTE "]*)" QUOTE ".*");
258  reg2.setCaseSensitivity(Qt::CaseInsensitive);
259  QRegExp reg3(".*(?:Etablissement).*(?:Nom) V.?=.?" QUOTE "([^" QUOTE "]+)" QUOTE ".*(?:Siret) V.?=.?" QUOTE "([0-9A-Z]+)" QUOTE);
260  reg3.setCaseSensitivity(Qt::CaseInsensitive);
261 
262 
263  QByteArray::const_iterator it;
264 
265  if (string.contains(reg))
266  {
267  Hash::Annee[fileName] = reg.cap(1);
268  Hash::Mois[fileName] = reg.cap(2);
269  QString budgetCapture = reg.cap(3);
270 
271  if (budgetCapture.contains(reg2))
272  Hash::Budget[fileName] = tools::remAccents(reg2.cap(1).replace("&#39;", "\'").replace("&apos;", "\'").trimmed());
273  else
274  Hash::Budget[fileName] = "" ;
275 
276  QString etabCapture = reg.cap(6);
277  if (etabCapture.contains(reg3))
278  Hash::Etablissement[fileName] << tools::remAccents(reg3.cap(1).replace("&#39;", "\'").replace("&apos;", "\'").trimmed());
279  else
280  Hash::Etablissement[fileName] << "" ;
281 
282  Hash::Employeur[fileName] = tools::remAccents(reg.cap(4).replace("&#39;", "\'").replace("&apos;", "\'").trimmed());
283 
284  if (etabCapture.contains(reg3))
285  {
286  Hash::Siret[fileName] << reg3.cap(2);
287  }
288  else
289  Hash::Siret[fileName] << reg.cap(5);
290  }
291  else
292  {
293  if (! string.toUpper().contains("DONNEESINDIV")) emit(textAppend(WARNING_HTML_TAG "Pas de données individuelles"));
294  if (! string.toUpper().contains("PAYEINDIVMENSUEL")) emit(textAppend(WARNING_HTML_TAG "Pas de payes individuelles"));
295 
296  emit(textAppend(WARNING_HTML_TAG "L'entête DocumentPaye... du fichier "
297  + fileName +
298  " est non conforme à l'annexe de la convention cadre de dématérialisation."));
299 
300  // DocumentPaye.*(?:Annee) V.?=.?" QUOTE "([0-9]+)" QUOTE ".*(?:Mois) V.?=.?" QUOTE "([0-9]+)" QUOTE "(.*)(?:Employeur).*(?:Nom) V.?=.?" QUOTE "([^" QUOTE "]+)" QUOTE ".*(?:Siret) V.?=.?" QUOTE "([0-9A-Z]+)" QUOTE ".*DonneesIndiv(.*)PayeIndivMensuel")
301 
302  Hash::Budget[fileName] = "";
303  Hash::Annee[fileName] = "Inconnu";
304  Hash::Mois[fileName] = "Inconnu";
305  Hash::Employeur[fileName] = "";
306  Hash::Etablissement[fileName] << "";
307  Hash::Siret[fileName] << "";
308 
309  /* effacer les fichiers mal formés de la liste des fichiers qui vont être envoyés en commandline */
310 
311  Hash::Suppression[fileName] = true;
312 
313  goto out;
314  }
315 
316  it = buffer0.cbegin() + string.indexOf("DonneesIndiv") + 15;
317 
318  while(it != buffer0.cend())
319  {
320 
321  if (*it == '\n')
322  {
323  ++ligne;
324  ++it;
325  continue;
326  }
327  if (*++it != '<') continue;
328  if (*++it != 'D') continue;
329  if (*++it != 'o') continue;
330  if (*++it != 'n') continue;
331  if (*++it != 'n') continue;
332  if (*++it != 'e') continue;
333  if (*++it != 'e') continue;
334  if (*++it != 's') continue;
335  if (*++it != 'I') continue;
336 
337  it += 6;
338 
339  for (int u = 0; u < BUFFER_SIZE; ++u) string[u] = *++it;
340 
341  QRegExp reg3("(?:Etablissement|Employeur).*(?:Nom) V=" QUOTE "([^" QUOTE "]+)" QUOTE ".*(?:Siret) V=" QUOTE "([0-9A-Z]+)" QUOTE "");
342  reg3.setPatternSyntax(QRegExp::RegExp2);
343 
344  if (string.contains(reg3))
345  {
346  QString s1 = reg3.cap(1).replace("&#39;", "\'");
347  QString s2 = reg3.cap(2);
348 
349  if (! Hash::Etablissement[fileName].contains(s1))
350  Hash::Etablissement[fileName] << s1;
351  if (! Hash::Etablissement[fileName].contains(s2))
352  Hash::Siret[fileName] << s2;
353  }
354  }
355 
356 
357 out :
358 
359  file.close();
360 
361  if (file.isOpen())
362  emit(textAppend(ERROR_HTML_TAG " Erreur à la fermeture du fichier."));
363 
364  if (file.error() != QFileDevice::NoError)
365  emit(textAppend(WARNING_HTML_TAG " Erreur de fichier."));
366 
367  emit(parsed());
368  return;
369 }

371 void FListFrame::parseXhlFile()
372 {

375 # endif
376  size = stringList.size();
377 
378  // L'usage de threads est indiqué pour éviter un freeze de l'interface graphique

380 
381  const QString cdROM = tools::cdRomMounted();
382 
383  if (isTerminated
384  || cdROM.isEmpty()
385  || ! QDir(cdROM).exists()
386  || QDir(cdROM).QDir::entryInfoList(QDir::Dirs
387  | QDir::Files
388  | QDir::NoDotAndDotDot).isEmpty())
389  {
390  for (auto &&s: stringList)
391  {
392  parseXhlFile(s);
393 # ifdef HAVE_APPLICATION
394  emit(setProgressBar(++rank));
395 # endif
396  }
397 
398  // N'utiliser des threads que si un disque optique est en input de données
399 
400  use_threads = false;
401 
402  importFromMainTree->show();
403  emit(imported());
404  return;
405  }
406 
407  isTerminated = false;
408  importFromMainTree->hide();
409  use_threads = true;

411 
412 #ifdef HAVE_APPLICATION

415 
416  launch_thread(0);
417 
418  #ifdef DEBUG_INPUT_FILES
419  app->outputTextEdit->append(PROCESSING_HTML_TAG "Analyse du fichier n°" + QString::number(rank));
420  #endif
421 
422  // Le thread contrôleur retarde l'importation des fichiers dans l'onglet central
423  // à la finalisation de la lecture du disue optique et gère aussi la barre
424  // de progression.
425 
426 
427  QTimer *timer = new QTimer(this);
428 
429  connect(timer, &QTimer::timeout, [&, timer] {
430  int test = 0;
431  for (QThread *t : thread)

435  }
436 
437  #ifdef HAVE_APPLICATION

439  #endif
440 
441  if (test == size)
442  {
443  emit(imported());
444  timer->stop();
445  }
446  });
447 
448  connect(this, SIGNAL(terminated()), timer, SLOT(stop()));
449  connect(this, &FListFrame::terminated, [this] {
450  isTerminated = true;
451  use_threads = false;
452  importFromMainTree->show();
453  });
454 
455  timer->start(PROGRESSBAR_TIMEOUT);
456 
457 }

486 QStringList FListFrame::parseTreeForFilePaths(const QStringList& stringList)
487 {
488 
489  QStringList stringsToBeAdded=QStringList();
490  QStringListIterator i(stringList);
491  if (importType != flags::importFiles) return QStringList();
492 
493  while (i.hasNext())
494  {
495  QString currentString = i.next();
496  if (currentString.isEmpty()) return QStringList();
497 
498  QFileInfo info = QFileInfo(currentString);
499  if (info.isDir())
500  {
501  QDir dir(currentString);
502  QFileInfoList entries = dir.entryInfoList(QDir::NoDotAndDotDot|QDir::Files|QDir::Dirs);
503  // Recursion
504  // Utilisation d'une rvalue
505  QStringList &&tempStrings=QStringList();
506  for (QFileInfo & embeddedFileInfo: entries)
507  {
508  tempStrings << embeddedFileInfo.absoluteFilePath();
509  }
510 
511  // Move semantics : gain de temps et de mémoire (>= Qt5.4)
512  // temStrings est coerced dans le type const QStringList sans copie de données
513 
514  stringsToBeAdded << parseTreeForFilePaths(tempStrings);
515 
516  }
517  else
518  if (info.isFile())
519  {
520  if (info.suffix().toUpper() == "XHL" || info.suffix().toUpper() == "XML")
521  stringsToBeAdded << currentString;
522 #ifdef HAVE_APPLICATION
523  else
524  emit(textAppend(WARNING_HTML_TAG + QString("Le fichier ")
525  + currentString + " sera ignoré. Les fichiers doivent avoir une extension du type .xml ou .xhl."));
526 
527 #endif
528  }
529  else return {};
530  }
531 
532  return stringsToBeAdded;
533 }
