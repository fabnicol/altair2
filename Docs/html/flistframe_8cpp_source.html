201 void FListFrame::launch_thread(int rank)
202 {
203  if (isTerminated || size == 0 || rank >= size) return;
204 
205  // Ajouter un fil d'exécution à la liste des fils thread
206 
207  thread.push_back(new QThread);
208 
209  const QString& fileName = stringList.at(rank);
210 
211  // Démarrer le fil et lui faire lire l'entête du fichier XHL en cours
212 
213  thread[rank]->start();
214  connect(thread[rank], &QThread::started, [this, fileName] {
215  parseXhlFile(fileName);
216  });
217 
218  // Lorsque le fil a fini de lire l'année, le mois etc. le signal parsed() est émis et le fil doit être arrêté

232 void FListFrame::parseXhlFile(const QString& fileName)
233 {
234 
235  QFile file(fileName);
236  long long ligne = 0;
237  bool result = file.open(QIODevice::ReadOnly);
238  if (! file.isOpen())
239  emit(textAppend(ERROR_HTML_TAG "Erreur à l'ouverture du fichier."));
240 
241  if (result == false || file.size()== 0)
242  {
243  emit(textAppend(WARNING_HTML_TAG "Fichier vide."));
244  return;
245  }
246 
247  file.seek(0);
248  QByteArray buffer0 = file.readAll();
249  QString string = QString::fromLatin1(buffer0, BUFFER_SIZE);
250 
251 # define QUOTE "(?:\"|')"
252 
253  QRegExp reg("DocumentPaye.*(?:Annee) V.?=.?" QUOTE "([0-9]+)" QUOTE ".*(?:Mois) V.?=.?" QUOTE "([0-9]+)" QUOTE "(.*)(?:Employeur).*(?:Nom) V.?=.?" QUOTE "([^" QUOTE "]+)" QUOTE ".*(?:Siret) V.?=.?" QUOTE "([0-9A-Z]+)" QUOTE ".*DonneesIndiv(.*)PayeIndivMensuel");
254  reg.setPatternSyntax(QRegExp::RegExp2);
255  reg.setCaseSensitivity(Qt::CaseInsensitive);
256  QRegExp reg2(".*Budget.*Libelle V.?=.?" QUOTE "([^" QUOTE "]*)" QUOTE ".*");
257  reg2.setCaseSensitivity(Qt::CaseInsensitive);
258  QRegExp reg3(".*(?:Etablissement).*(?:Nom) V.?=.?" QUOTE "([^" QUOTE "]+)" QUOTE ".*(?:Siret) V.?=.?" QUOTE "([0-9A-Z]+)" QUOTE);
259  reg3.setCaseSensitivity(Qt::CaseInsensitive);
260 
261  QByteArray::const_iterator it;
262 
263  if (string.contains(reg))
264  {
265  Hash::Annee[fileName] = reg.cap(1);
266  Hash::Mois[fileName] = reg.cap(2);
267  QString budgetCapture = reg.cap(3);
268 
269  if (budgetCapture.contains(reg2))
270  Hash::Budget[fileName] = tools::remAccents(reg2.cap(1).replace("&#39;", "\'").replace("&apos;", "\'").trimmed());
271  else
272  Hash::Budget[fileName] = "" ;
273 
274  QString etabCapture = reg.cap(6);
275  if (etabCapture.contains(reg3))
276  Hash::Etablissement[fileName] << tools::remAccents(reg3.cap(1).replace("&#39;", "\'").replace("&apos;", "\'").trimmed());
277  else
278  Hash::Etablissement[fileName] << "" ;
279 
280  Hash::Employeur[fileName] = tools::remAccents(reg.cap(4).replace("&#39;", "\'").replace("&apos;", "\'").trimmed());
281 
282  if (etabCapture.contains(reg3))
283  {
284  Hash::Siret[fileName] << reg3.cap(2);
285  }
286  else
287  Hash::Siret[fileName] << reg.cap(5);
288  }
289  else
290  {
291  if (! string.toUpper().contains("DONNEESINDIV")) emit(textAppend(WARNING_HTML_TAG "Pas de données individuelles"));
292  if (! string.toUpper().contains("PAYEINDIVMENSUEL")) emit(textAppend(WARNING_HTML_TAG "Pas de payes individuelles"));
293 
294  emit(textAppend(WARNING_HTML_TAG "L'entête DocumentPaye... du fichier "
295  + fileName +
296  " est non conforme à l'annexe de la convention cadre de dématérialisation."));
297 
298  // DocumentPaye.*(?:Annee) V.?=.?" QUOTE "([0-9]+)" QUOTE ".*(?:Mois) V.?=.?" QUOTE "([0-9]+)" QUOTE "(.*)(?:Employeur).*(?:Nom) V.?=.?" QUOTE "([^" QUOTE "]+)" QUOTE ".*(?:Siret) V.?=.?" QUOTE "([0-9A-Z]+)" QUOTE ".*DonneesIndiv(.*)PayeIndivMensuel")
299 
300  Hash::Budget[fileName] = "";
301  Hash::Annee[fileName] = "Inconnu";
302  Hash::Mois[fileName] = "Inconnu";
303  Hash::Employeur[fileName] = "";
304  Hash::Etablissement[fileName] << "";
305  Hash::Siret[fileName] << "";
306 
307  /* effacer les fichiers mal formés de la liste des fichiers qui vont être envoyés en commandline */
308 
309  Hash::Suppression[fileName] = true;
310 
311  goto out;
312  }
313 
314  it = buffer0.cbegin() + string.indexOf("DonneesIndiv") + 15;
315 
316  while(it != buffer0.cend())
317  {
318 
319  if (*it == '\n')
320  {
321  ++ligne;
322  ++it;
323  continue;
324  }
325  if (*++it != '<') continue;
326  if (*++it != 'D') continue;
327  if (*++it != 'o') continue;
328  if (*++it != 'n') continue;
329  if (*++it != 'n') continue;
330  if (*++it != 'e') continue;
331  if (*++it != 'e') continue;
332  if (*++it != 's') continue;
333  if (*++it != 'I') continue;
334 
335  it += 6;
336 
337  for (int u = 0; u < BUFFER_SIZE; ++u) string[u] = *++it;
338 
339  QRegExp reg3("(?:Etablissement|Employeur).*(?:Nom) V=" QUOTE "([^" QUOTE "]+)" QUOTE ".*(?:Siret) V=" QUOTE "([0-9A-Z]+)" QUOTE "");
340  reg3.setPatternSyntax(QRegExp::RegExp2);
341 
342  if (string.contains(reg3))
343  {
344  QString s1 = reg3.cap(1).replace("&#39;", "\'");
345  QString s2 = reg3.cap(2);
346 
347  if (! Hash::Etablissement[fileName].contains(s1))
348  Hash::Etablissement[fileName] << s1;
349  if (! Hash::Etablissement[fileName].contains(s2))
350  Hash::Siret[fileName] << s2;
351  }
352  }
353 
354 
355 out :
356 
357  file.close();
358 
359  if (file.isOpen())
360  emit(textAppend(ERROR_HTML_TAG " Erreur à la fermeture du fichier."));
361 
362  if (file.error() != QFileDevice::NoError)
363  emit(textAppend(WARNING_HTML_TAG " Erreur de fichier."));
364 
365  emit(parsed());
366  return;
367 }

369 void FListFrame::parseXhlFile()
370 {
371 # ifdef HAVE_APPLICATION
372  int rank = 0;
373 # endif
374  size = stringList.size();
375 
376  // L'usage de threads est indiqué pour éviter un freeze de l'interface graphique
377  // pendant que le disque optique est parcouru pour examiner les entêtes
378 
379  const QString cdROM = tools::cdRomMounted();
380 
381  if (isTerminated
382  || cdROM.isEmpty()
383  || ! QDir(cdROM).exists()
384  || QDir(cdROM).QDir::entryInfoList(QDir::Dirs
385  | QDir::Files
386  | QDir::NoDotAndDotDot).isEmpty())
387  {
388  for (auto &&s: stringList)
389  {
390  parseXhlFile(s);
391 # ifdef HAVE_APPLICATION
392  emit(setProgressBar(++rank));
393 # endif
394  }
395 
396  // N'utiliser des threads que si un disque optique est en input de données
397 
398  use_threads = false;
399 
400  importFromMainTree->show();
401  emit(imported());
402  return;
403  }
404 
405  isTerminated = false;
406  importFromMainTree->hide();
407  use_threads = true;
408  if (size == 0) return;
409 
410 #ifdef HAVE_APPLICATION
411  emit(showProgressBar());
412 #endif
413 
414  launch_thread(0);
415 
416  #ifdef DEBUG_INPUT_FILES
417  app->outputTextEdit->append(PROCESSING_HTML_TAG "Analyse du fichier n°" + QString::number(rank));
418  #endif
419 
420  // Le thread contrôleur retarde l'importation des fichiers dans l'onglet central
421  // à la finalisation de la lecture du disue optique et gère aussi la barre
422  // de progression.
423 
424 
425  QTimer *timer = new QTimer(this);
426 
427  connect(timer, &QTimer::timeout, [&, timer] {
428  int test = 0;
429  for (QThread *t : thread)
430  {
431  if (t)
432  test += (int) t->isFinished() ;
433  }
434 
435  #ifdef HAVE_APPLICATION
436  emit(setProgressBar(test));
437  #endif
438 
439  if (test == size)
440  {
441  emit(imported());
442  timer->stop();
443  }
444  });
445 
446  connect(this, SIGNAL(terminated()), timer, SLOT(stop()));
447  connect(this, &FListFrame::terminated, [this] {
448  isTerminated = true;
449  use_threads = false;
450  importFromMainTree->show();
451  });
452 
453  timer->start(PROGRESSBAR_TIMEOUT);
454 
455 }

484 QStringList FListFrame::parseTreeForFilePaths(const QStringList& stringList)
485 {
486 
487  QStringList stringsToBeAdded=QStringList();
488  QStringListIterator i(stringList);
489  if (importType != flags::importFiles) return QStringList();
490 
491  while (i.hasNext())
492  {
493  QString currentString = i.next();
494  if (currentString.isEmpty()) return QStringList();
495 
496  QFileInfo info = QFileInfo(currentString);
497  if (info.isDir())
498  {
499  QDir dir(currentString);
500  QFileInfoList entries = dir.entryInfoList(QDir::NoDotAndDotDot|QDir::Files|QDir::Dirs);
501  // Recursion
502  // Utilisation d'une rvalue
503  QStringList &&tempStrings=QStringList();
504  for (QFileInfo & embeddedFileInfo: entries)
505  {
506  tempStrings << embeddedFileInfo.absoluteFilePath();
507  }
508 
509  // Move semantics : gain de temps et de mémoire (>= Qt5.4)
510  // temStrings est coerced dans le type const QStringList sans copie de données
511 
512  stringsToBeAdded << parseTreeForFilePaths(tempStrings);
513 
514  }
515  else
516  if (info.isFile())
517  {
518  if (info.suffix().toUpper() == "XHL" || info.suffix().toUpper() == "XML")
519  stringsToBeAdded << currentString;
520 #ifdef HAVE_APPLICATION
521  else
522  emit(textAppend(WARNING_HTML_TAG + QString("Le fichier ")
523  + currentString + " sera ignoré. Les fichiers doivent avoir une extension du type .xml ou .xhl."));
524 
525 #endif
526  }
527  else return {};
528  }
529 
530  return stringsToBeAdded;
531 }
