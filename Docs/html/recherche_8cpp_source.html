50 vector<string> recherche (const vector<info_t> &Info, const string& annee, const string& mois, const string& matricule)
51 {
52  // Bulletins à extraire
53  vector<string> bulletins;
54 
55  auto matr = (const xmlChar*) matricule.c_str();
56  int m = stoi (mois);
57  int a = stoi (annee);
58 
59  // Parcourt l'ensemble des données de paye, pour tous les fils d'exécution, après décodage
60  // Il faut donc que lhx se soit complètement exécuté préalablement
61 
62  // Boucle sur les fils d'exécution
63 
64  for (unsigned int i = 0; i < Info[0].nbfil; ++i)
65  {
66 
67  // Boucle sur les données extraites pour un fil donné
68 
69  for (vector<vector<xmlChar*>>::const_iterator it = Info[i].Table.begin(); it != Info[i].Table.end(); ++it)
70  {
71  // On restreint la recherche à l'année, au mois et au matricule donnés
72  // On pourrait aller plus vite avec une table de hachage, mais l'expérience montre que ce n'est pas nécessaire
73  // it correspond à la partie de la Table pour un agent donné
74 
75  if (atoi ((const char*) it->at (Annee)) == a
76  && atoi ((const char*) it->at (Mois)) == m
77  && xmlStrcmp (it->at (Matricule), matr) == 0)
78  {
79  // index correspond au rang de l'agent dans la Table (0 <= index <= NCumAgentXml)
80 
81  long long index = it - Info[i].Table.begin();
82 
83  // le vecteur ligne_debut a été construit lors de l'extraction : il permet de situer la ligne de début
84  // du code XML correspondant à l'agent de rang index dans le fichier et donnant le début de son propre bulletin
85  // de paye au format XML
86  // le vecteur ligne_fin donne la ligne de fin de ce code-bulletin
87 
88  array<uint64_t, 3> debut = Info[i].ligne_debut.at (index);
89  array<uint64_t, 2> fin = Info[i].ligne_fin.at (index);
90 
91  // Lancer la fonction extraire_ligne qui copie les lignes XML entre la ligne de début et la ligne de fin
92 
93  const string fichier = extraire_lignes (Info[i], debut, fin);
94 
95  // Cela ne suffit pas à donner un fichier XML syntaxiquement correct.
96  // A cette fin, rajouter un préambule et une fin de fichier en accord avec ce préambule.
97  // Ce qui permettra éventuellement de "repasser" ce bulletin XML artificiellement créé dans lhx
98  // pour extraire ce bulletin particulier au format CSV
99 
100  const string preambule =
101  "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n\
102 <DocumentPaye>\n\
103  <IdVer V=\"1.0\"/>\n\
104  <Annee V=\"" + annee + "\"/>\n\
105  <Mois V=\"" + mois + "\"/>\n\
106  <Budget>\n\
107  <Libelle V=\"" + string ((const char*)it->at (Budget)) + "\"/>\n\
108  <Code V=\"\"/>\n\
109  </Budget>\n\
110  <Employeur>\n\
111  <Nom V=\"" + string ((const char*) it->at (Employeur)) + "\"/>\n\
112  <Siret V=\"" + string ((const char*) it->at (Siret)) + "\"/>\n\
113  </Employeur>\n\
114  <DonneesIndiv>\n";
115 
116  const string coda = " </DonneesIndiv>\n</DocumentPaye>\n";
117 
118  // Empiler le tout dans le vecteur bulletins
119 
120  if (! fichier.empty())
121  bulletins.emplace_back (preambule + fichier + coda);
122  }
123  }
124 
125  }
126 
127  return bulletins;
128 }

130 const string extraire_lignes (const info_t& info, const array<uint64_t, 3>& debut, const array <uint64_t, 2>& fin)
131 {
132 
133  if (fin[1] < debut[1]) return "";
134 
135  string xhl = info.threads->in_memory_file.at (debut[2]);
136 
137  string tab = xhl.substr (debut[1], fin[1] - debut[1] + 1);
138 
139  tab += "\n";
140

144 vector<string> bulletin_paye (const string& chemin_repertoire, const vector<info_t> &Info, const string& matricule, const string& mois, const string& annee)
145 {
146  cerr << ENDL;
147  cerr << PROCESSING_HTML_TAG << "Extraction des bulletins..." << ENDL;
148 
149  auto bulletins = recherche (Info, annee, mois, matricule);
150  int rang = 0;
151  vector<string> chemins_bulletins;
152 
153  for (auto &&bulletin : bulletins)
154  {
155  ++rang;
156 
157  // Classement par Matricule - Année - Mois
158  // Cet ordre permet un lcassement automatique optimal dans les gestionnaires de fichiers des SE.
159  // Dans la pratique en effet on recherche d'abord les bulletins d'un agent, en triant ensuite par exercice et par mois.
160 
161  string nom_bulletin = "Paye-" + matricule + "-" + annee + "-" + mois + ".xml";
162 
163  ofstream f;
164 
165  string chemin_bulletin = chemin_repertoire + "/" + nom_bulletin;
166 
167  chemins_bulletins.emplace_back(chemin_bulletin);
168 
169  f.open (chemin_bulletin, std::ofstream::out | std::ofstream::trunc);
170 
171  if (f.is_open())
172  f << bulletin;
173  else
174  {
175  cerr << ERROR_HTML_TAG "Echec de la génération du bulletin pour le matricule "
176  << matricule << " Année : " << annee << " Mois : " << mois << ENDL;
177 
178  continue;
179  }
180 
181  f.close();
182 
183  cerr << STATE_HTML_TAG "Bulletin extrait pour le matricule " << matricule;
184 
185  if (rang > 1)
186  {
187  cerr << " Rang local : " << rang;
188  }
189 
190  cerr << " Année : " << annee << " Mois : " << mois << ENDL;
191  }
192 
193  return chemins_bulletins;
194 }

197 vector<string> __scan_mois (const string &repertoire_bulletins,
198  const vector<info_t> &Info,
199  const string &matricule,
200  const string &mois,
201  const string &annee)
202 {
203  size_t pos = 0;
204  vector<string>chemins_bulletins_extraits;
205 
206  // Les mois peuvent être donnés en intervalles du type 02...11
207  // ce qui signifie : tous les mois entre février et novembre inclus
208 
209  if ((pos = mois.find_first_of ('.')) != string::npos)
210  {
211  int m0 = stoi (mois.substr (0, pos));
212  pos = mois.find_last_of ('.');
213  int m1 = stoi (mois.substr (pos + 1));
214 
215  // Boucler entre les deux mois ainsi donnés en borne inf et max
216  // et lancer la fonction bulletin_paye sur chacun de ces mois
217 
218  for (int m = m0; m <= m1; ++m)
219  {
220  if (verbeux) cerr << PROCESSING_HTML_TAG "Année : " << annee << " Extraction du mois : " << m << ENDL;
221 
222  vector<string> c = bulletin_paye (repertoire_bulletins,
223  Info,
224  matricule,
225  to_string (m),
226  annee);
227 
228  vect_concat(chemins_bulletins_extraits, c);
229  }
230  }
231 
232  // Si pas d'intervalle, lancer la fonction bulletin_paye sur le seul mois donné.
233 
234  else
235  {
236  if (verbeux) cerr << PROCESSING_HTML_TAG "Année : " << annee << " Extraction du mois : " << mois << ENDL;
237  chemins_bulletins_extraits = bulletin_paye (repertoire_bulletins,
238  Info,
239  matricule,
240  mois,
241  annee);
242 
243  }
244 
245  return chemins_bulletins_extraits;
246 }

252  const string &annee)

256 
257 // Les matricules peuvent être séparés par des virgules
258 
259 if ((pos = matricule.find_first_of (',')) != string::npos)
260  {
261  vector<string> matriculeList = split(matricule, ',');
262 
263  for (auto &&m : matriculeList)
264  {
265  vect_concat(chemins_bulletins_extraits,
266  __scan_mois (repertoire_bulletins,
267  Info,
268  m,
269  mois,
270  annee));
271  }
272  }
273 else
274  chemins_bulletins_extraits = __scan_mois (repertoire_bulletins,
275  Info,
276  matricule,
277  mois,
278  annee);
279 
280 return chemins_bulletins_extraits;
281 }
