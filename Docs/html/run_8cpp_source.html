62 QStringList Altair::createCommandLineString (const QStringList& files)
63 {
64  // liste des composant fonctionnels
65 
66  QVectorIterator<FAbstractWidget*> w (Abstract::abstractWidgetList);
67  QStringList commandLine;
68 
69  w.toBack();
70  fileCount = 0;
71 
72  while (w.hasPrevious())
73  {
74  FAbstractWidget* item = w.previous();
75  QStringList commandLineChunk;
76 
77  // Les onglets correspondent au fwidget FListFrame de balise "XHL"
78 
79  if (! files.isEmpty() && item->getHashKey() == "XHL")
80  {
81  commandLineChunk = files;
82  }
83  else
84  {
85  if (item->getHashKey() != "dossierBulletins"
86  || v(dossierBulletins).isEmpty()
87  || v(Matricules).isFilled()
88  || v(MatriculesB).isFilled()
89  || v(MatriculesC).isFilled())
90 
91  // Cas général : on récupère le bout de ligne de commande généré par chaque fwidget :
92 
93  commandLineChunk = item->commandLineStringList();
94  }
95 
96  // Certains fidgets génèrent une ligne de commande
97  // C'est le cas pour celui de balise "exportMode" mais la ligne de doit pas être crée ici :
98  // On saute ce cas :
99 
100  if (item->getHashKey() == "exportMode")
101  {
102  continue;
103  }
104 
105  // Collage des bouts de ligne de commande :
106 
107  if (! commandLineChunk.isEmpty() && ! commandLineChunk[0].isEmpty())
108  commandLine += commandLineChunk;
109 
110  // Le nombre de fichiers de paye filecount est simplement égal à la taille de la liste commandLineChunk
111 
112  if (item->getHashKey() == "XHL")
113  {
114  fileCount = commandLineChunk.size();
115 
116  if (fileCount == 0) return QStringList();
117  }
118  }
119 
120  return commandLine;
121 }

385 void Altair::run()
386 {
387  updateProject (true);
388 
389  // crucial otherwise some dynamic settings in the option dialog
390  // may not get through to command line
391 
392  if (Altair::totalSize == 0)
393  {
394  processFinished (exitCode::shouldLaunchRAltairAlone);
395  return;
396  }
397 
398  if (Hash::wrapper.isEmpty() || Hash::wrapper["XHL"] == nullptr
399  || Hash::wrapper["XHL"]->isEmpty() || Hash::wrapper["XHL"]->at (0).isEmpty()
400  || Hash::wrapper["XHL"]->at (0).at (0).isEmpty())
401  {
402  QMessageBox::warning (nullptr, "Projet", "Charger un projet !");
403  processFinished (exitCode::shouldLaunchRAltairAlone);
404  return;
405  }
406 
407  QString path = v(base);
408 
409  if (path.isEmpty())
410  {
411  Warning ("Répertoire de sortie", "Le répertoire de création des bases " + path +
412  " n'a pas été indiqué, renseigner le dialogue des paramètres.");
413 
414  processFinished (exitCode::shouldLaunchRAltairAlone);
415  return;
416  }
417 
418  QDir targetDirObject (path);
419 
420  if (! targetDirObject.exists() && ! targetDirObject.mkpath (path))
421  {
422  QMessageBox::critical (nullptr, "Répertoire des bases", "Le répertoire " + path +
423  " n'a pas pu être créé. Veuillez le faire manuellement.");
424 
425  processFinished (exitCode::shouldLaunchRAltairAlone);
426  return;
427  }
428 
429  const QStringList& files = targetDirObject.entryList (QDir::Files
430  | QDir::Dirs
431  | QDir::NoDotAndDotDot);
432 
433  if (v(genererTable).isTrue() && ! files.isEmpty() && v(exportMode) != "Cumulative" && v(exportMode) != "Distributive+")
434  {
435  if (QMessageBox::Cancel
436  == QMessageBox::warning (this, QString ("Attention"),
437  tr ("Vous allez supprimer les fichiers créés par le précédent traitement.\n"),
438  QMessageBox::Ok | QMessageBox::Cancel))
439  {
440  processFinished (exitCode::shouldLaunchRAltairAlone);
441  return;
442  }
443 
444  for (const QString& file : files)
445  {
446  const QString filepath = path + "/" + file;
447 
448  if (QFileInfo (filepath).isFile())
449  {
450  QFile::remove (filepath);
451  }
452  else
453  {
454  QDir (filepath).removeRecursively();
455  }
456  }
457  }
458 
459  Hash::fileList.clear();
460 
461  if (v(exportMode).left (12) == "Distributive")
462  {
463  const QString cdROM = cdRomMounted();
464 
465  for (int j = 0; j < Hash::wrapper["XHL"]->size() - 3; ++j)
466  {
467  const QStringList &q = Hash::wrapper["XHL"]->at (j);
468 
469  for (const QString &s : q)
470  {
471  QString d = s.section ("xhl/", 1, 1, QString::SectionSkipEmpty);
472  bool hasSubDir = false;
473 # ifdef Q_OS_WIN
474 
475  hasSubDir = d.count ('/') > 0;
476  d = d.section ('/', 0, 0, QString::SectionSkipEmpty);
477 
478 # else
479 
480  if (username == "fab")
481  {
482  hasSubDir = d.count ('/') > 0;
483  d = d.section ('/', 0, 0, QString::SectionSkipEmpty);
484  }
485  else
486  {
487  hasSubDir = d.count ('/') > 1;
488  d = d.section ('/', 1, 1, QString::SectionSkipEmpty);
489  }
490 
491 # endif
492 
493  if (d.isEmpty() && ! cdROM.isEmpty())
494  {
495  d = s.section (cdROM, 0, 0, QString::SectionSkipEmpty);
496  hasSubDir = d.count ('/') > 0;
497 
498  d = d.section ('/', 0, 0, QString::SectionSkipEmpty);
499  rootDir = cdROM;
500  }
501  else
502  rootDir = userdatadir;
503 
504  const QString &outdir = v(base) + QDir::separator() + d;
505 
506  if (hasSubDir && ! d.isEmpty() && ! QFileInfo (outdir).isDir())
507  {
508  QDir().mkpath (outdir);
509  }
510 
511  Hash::fileList[hasSubDir ? d : ""] << s;
512  }
513  }
514 
515  if (! Hash::fileList.empty())
516  {
517  runWorkerDistributed (true);
518  }
519  }
520  else
521  {
522  common::exporter_identification_controle();
523  runWorker();
524  }
525 }
