294 static inline void generate_rank_signal()
295 {
296  if (rankFilePath.empty()) return;
297 
298  while (! mut.try_lock()) {}
299 
300  static int temp_rank;
301 
302  do
303  {
304  rankFile.open (rankFilePath, ios::out | ios::trunc);
305 
306  if (rankFile.is_open())
307  {
308  if (rang_global)
309  rang_global = temp_rank;
310 
311  rankFile << ++rang_global ;
312  temp_rank = rang_global;
313  }
314 
315  rankFile.close();
316 
317 
318  mut.unlock();
319 
320  }
321  while (false);
322 
323 }
