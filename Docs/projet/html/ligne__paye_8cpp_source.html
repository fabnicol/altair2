109 static void GCC_INLINE sanitize(xmlChar* s, const char sep)
110 {
111 
112  while (*s != 0)
113  {
114  // Non-switchable car info.seperateur n'est pas une expression constante.
115  if (*s == sep) *s = '_';
116 
117  switch(*s)
118  {
119 
120  case '\n':
121  *s = ' ';
122  break;
123 
124 #ifdef CONVERTIR_LATIN_1
125 #if !defined(__linux__) && ! defined(USE_ICONV)
126 
127 // Gros hack de pseudo-conversion UTF-8 vers Latin-1, qui permet d'économiser les 40 %
128 // de surcoût d'exécution lié à l'utilisation d'iconv pour retraiter les fichiers de
129 // sortie (fonction convertir(const char*))
130 // Ce hack est presque sans coût. Il se base sur les hypothèses suivantes :
131 // a) pas de caractères spéciaux multioctets
132 // b) seuls sont convertis : à, â, ç, è, é, ê, ë, î, ï, ô, û ... et les majuscules
133 // correspondantes càd
134 // dont le code UTF-8 commence par 0xC3. Il suffit d'ajouter 0x40 sur les quatre bits
135 // hauts de l'octet.
136 
137  case 0xC3:
138 
139  *s = ((*(s + 1) & 0xF0) + 0x40) | (*(s + 1) & 0x0F);
140 
141  effacer_char(s + 1);
142 
143  break;
144 
145  case 0xC2:
146 
147  *s = *(s + 1);
148 // Le caractère ° (degré) est bien codé en Latin-1 comme 0xB0, mais il y a un
149 // problème avec le paquet texlive inputenc pour la conversion pdf.
150 // On remplace donc par (0x65). Apparemment plus nécessaire */
151 // if (info.Table[info.NCumAgentXml][l][i] == 0xB0)
152 // info.Table[info.NCumAgentXml][l][i] = 0x65;
153 
154  effacer_char(s + 1);
155 
156  break;
157 #endif
158 #endif
159  }
160 
161  ++s;
162  }
163 }

661 static inline LineCount lignePaye(xmlNodePtr cur, info_t& info)
662 {
663  int l = BESOIN_MEMOIRE_ENTETE;
664 
665  int nbLignePaye = 0;
666 
667  unsigned int t = 0;
668 
669 // +1 pour éviter la confusion avec \0 des chaines vides
670  info.Table[info.NCumAgentXml][l] = (xmlChar*) xmlStrdup(drapeau[t]);
671  ++l;
672 
673 // Besoins en mémoire :
674 // BESOIN_MEMOIRE_ENTETE [champs hors ligne] + nombre de lignes
675 // + flags (maximum nbType * nb de rembobinages)
676 
677  int type_loop_counter = 0;
678 
679  while (cur != nullptr)
680  {
681  bool new_type = false;
682 
683  while (xmlStrcmp(cur->name, (const xmlChar *) type_remuneration[t]))
684  {
685  // Cas rare dans lequel <Remuneration> n'existe pas
686  if (xmlStrcmp(cur->name, (const xmlChar *) "NbHeureTotal") == 0)
687  return { nbLignePaye, l};
688 
689  ++t;
690  if (t == nbType)
691  {
692 // En principe les éléments constitutifs des enregistrements
693 // <Remunération>....</Remuneration> sont enregistrés
694 // dans l'ordre du tableau type_remuneration. Toutefois quelques cas de
695 // désordre sont observés. Dans ces cas là on peut
696 // "rembobiner le tableau". On évite toutefois de faire une recherche
697 // ensembliste systématique, qui éviterait cela mais
698 // freinerait 99,9 % des recherches
699 
700  if (++type_loop_counter < TYPE_LOOP_LIMIT)
701  {
702  t = 0;
703  continue;
704  }
705 
706 // On ne rembobine qu'au maximum TYPE_LOOP_LIMIT. Si l'essai échoue,
707 // on déclenche une exception ou on retourne
708 
709  if (cur)
710  cerr << ERROR_HTML_TAG "Type litigieux " << cur->name
711  << " aux alentours du matricule "
712  << info.Table[info.NCumAgentXml][Matricule] << ENDL;
713  else
714  cerr << ERROR_HTML_TAG "Pointeur noeud courant nul" << ENDL;
715 
716  #ifdef STRICT
717  exit(-11);
718  #else
719  cerr << ERROR_HTML_TAG "Arrêt du décodage de la ligne de paye." << ENDL;
720  return {nbLignePaye, l};
721  #endif
722  }
723 
724  new_type = true;
725  }
726 
727  if (new_type && t < nbType)
728  {
729 // +1 pour éviter la confusion avec \0 des chaines vides
730  if ((info.Table[info.NCumAgentXml][l]
731  = (xmlChar*) xmlStrdup(drapeau[t])) == nullptr)
732  {
733  LOCK_GUARD
734  if (verbeux) cerr << ERROR_HTML_TAG "Erreur dans l'allocation des drapeaux de catégories." << ENDL;
735  #ifdef STRICT
736  exit(-12);
737  #else
738  if (verbeux) cerr << ERROR_HTML_TAG "Arrêt du décodage de la ligne de paye." << ENDL;
739  #endif
740  return {nbLignePaye, l};
741  }
742 
743  ++l;
744  }
745 
746 // ici on pourrait in fine se passer de ce test par compilation séparée
747 
748  if (! info.reduire_consommation_memoire)
749  {
750  verifier_taille(nbLignePaye, info);
751  }
752 
753 // Si on arrive à un noeud de type Commentaire, on le saute et on réinitialise
754 // "gratuitement" le parcours des drapeaux.
755 
756  if (xmlStrcmp(cur->name, (const xmlChar*) "Commentaire") == 0)
757  {
758  cur = cur->next;
759  t=0;
760  --type_loop_counter; // 'Rembobinage gratuit'
761 
762  continue; // garantit incidemment que cur != nullptr dans la boucle
763  }
764 
765 // cur n'est pas nul à ce point
766 
767  cur = cur->xmlChildrenNode;
768 
769  if (cur == nullptr)
770  {
771  for (short c = 0; c < 6; ++c) NA_ASSIGN(l++);
772  ++nbLignePaye;
773 
774  if (verbeux) cerr << WARNING_HTML_TAG
775  "Anomalie : la ligne de paye est vide."
776  << ENDL;
777  break;
778  }
779 
780 // cur n'est pas nul à ce point et ne devient jamais nul ci-après
781 
782 // Libellé, obligatoire
783 
784  bulletin_obligatoire("Libelle", cur, l, info);
785 
786  ++l;
787 
788 // Code, obligatoire
789 
790  bulletin_obligatoire("Code", cur, l, info);
791 
792  ++l;
793 
794 // Base, si elle existe
795 
796  bulletin_optionnel_numerique("Base", cur, l, info);
797 
798  ++l;
799 
800 // Taux, s'il existe
801 
802  bulletin_optionnel_numerique("Taux", cur, l, info);
803 
804  ++l;
805 
806 // Nombre d'unités, s'il existe
807 
808  bulletin_optionnel_numerique("NbUnite", cur, l, info);
809 
810  ++l;
811 
812 // Montant obligatoire
813 
814 
815  if (t == 8) // La catégorie de ligne de paye est "Rappel"
816  {
817  if (cur->xmlChildrenNode == nullptr)
818  {
819  if (verbeux)
820  {
821  LOCK_GUARD
822  cerr << WARNING_HTML_TAG "Pas de période de référence pour le rappel" " pour le matricule " << info.Table[info.NCumAgentXml][Matricule] << " -- Ligne";
823  long lineN = xmlGetLineNo(cur);
824  if (lineN != 65535)
825  {
826  cerr << " " << lineN << ENDL;
827  }
828  else
829  {
830  if (info.ligne_debut.size() > info.NCumAgentXml
831  && info.ligne_fin.size() > info.NCumAgentXml)
832  {
833  cerr << "s " << info.ligne_debut.at(info.NCumAgentXml)[0] + 1
834  << " - " << info.ligne_fin.at(info.NCumAgentXml)[0] + 1 << ENDL;
835  }
836  }
837  }
838  NA_ASSIGN(l);
839  NA_ASSIGN(++l);
840  }
841  else
842  {
843  cur = cur->xmlChildrenNode;
844  // On ne tient pas rigueur du manque de qualité éventuelle
845  // tellement la norme est peu respectée
846 
847  bulletin_optionnel_char("DateDebut", cur, l, info);
848 
849  ++l;
850  info.drapeau_cont = false; // pas de noeud successeur
851  bulletin_optionnel_char("DateFin", cur, l, info);
852  cur = cur->parent->next;
853  info.drapeau_cont = true; // noeud successeur existe
854  }
855  }
856  else
857  {
858  NA_ASSIGN(l);
859  NA_ASSIGN(++l);
860  }
861 
862  ++l;
863 // cur ne sera pas nul à ce point
864 
865  ++nbLignePaye;
866 
867  cur = cur->parent->next;
868 
869 // Le parent ne peut être nul
870 // attention si du code est rajouté ici il doit l'être sous garde cur != nullptr
871 // Lorsque on a épuisé tous les types licites on a nécessairement cur = nullptr et
872 // la boucle s'arrête
873 
874  }
875 
876 
877  return { nbLignePaye, l};
878 }
